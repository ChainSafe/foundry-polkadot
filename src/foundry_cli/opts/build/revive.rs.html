1use clap::Parser;
2use foundry_config::{SolcReq, revive::ResolcConfig};
3use serde::Serialize;

5#[derive(Clone, Debug, Default, Serialize, Parser)]
6#[clap(next_help_heading = "Resolc configuration")]
7/// Compiler options for resolc
8pub struct ResolcOpts {
9    #[arg(
10        value_name = "RESOLC_COMPILE",
11        help = "Enable compiling with resolc",
12        long = "resolc-compile",
13        visible_alias = "resolc",
14        action = clap::ArgAction::SetTrue
15    )]
16    pub resolc_compile: Option<bool>,
17
18    /// Enable PVM mode at startup (independent of compilation)
19    #[arg(
20        long = "resolc-startup",
21        help = "Enable PVM mode at startup",
22        value_name = "RESOLC_STARTUP",
23        action = clap::ArgAction::SetTrue
24    )]
25    pub resolc_startup: Option<bool>,
26
27    /// Specify the resolc version, or a path to a local resolc, to build with.
28    ///
29    /// Valid values follow the SemVer format `x.y.z-dev.n`, `resolc:x.y.z-dev.n` or
30    /// `path/to/resolc`.
31    #[arg(
32        long = "use-resolc",
33        help = "Use resolc version",
34        alias = "resolc-compiler-version",
35        value_name = "RESOLC_VERSION"
36    )]
37    #[serde(skip)]
38    pub use_resolc: Option<String>,
39
40    /// Set the LLVM optimization parameter `-O[0 | 1 | 2 | 3 | s | z]`.
41    /// Use `3` for best performance and `z` for minimal size.
42    #[arg(
43        short = 'O',
44        long = "resolc-optimizer-mode",
45        help = "Set the resolc optimization mode `-O[0 | 1 | 2 | 3 | s | z]`",
46        visible_alias = "resolc-optimization",
47        value_name = "LEVEL"
48    )]
49    #[serde(skip_serializing_if = "Option::is_none")]
50    pub optimizer_mode: Option<String>,
51
52    /// The emulated EVM linear heap memory static buffer size in bytes.
53    #[arg(long = "heap-size", help = "Set the contracts heap size in bytes", value_name = "SIZE")]
54    #[serde(skip_serializing_if = "Option::is_none")]
55    pub heap_size: Option<u32>,
56
57    /// The contracts total stack size in bytes.
58    #[arg(
59        long = "stack-size",
60        help = "Set the contracts total stack size in bytes",
61        value_name = "SIZE"
62    )]
63    #[serde(skip_serializing_if = "Option::is_none")]
64    pub stack_size: Option<u32>,
65
66    /// Generate source based debug information in the output code file.
67    #[arg(
68        long = "debug-info",
69        help = "Generate source based debug information in the output code file",
70        action = clap::ArgAction::SetTrue
71    )]
72    #[serde(skip_serializing_if = "Option::is_none")]
73    pub debug_information: Option<bool>,
74}
75
76impl ResolcOpts {
77    pub(crate) fn apply_overrides(&self, mut resolc: ResolcConfig) -> ResolcConfig {
78        macro_rules! set_if_some {
79            ($src:expr, $dst:expr) => {
80                if let Some(src) = $src {
81                    $dst = src.into();
82                }
83            };
84        }
85
86        set_if_some!(
87            self.resolc_compile.and_then(|v| if v { Some(true) } else { None }),
88            resolc.resolc_compile
89        );
90        set_if_some!(
91            self.resolc_startup.and_then(|v| if v { Some(true) } else { None }),
92            resolc.resolc_startup
93        );
94        set_if_some!(
95            self.use_resolc.as_ref().map(|v| SolcReq::from(v.trim_start_matches("resolc:"))),
96            resolc.resolc
97        );
98        set_if_some!(
99            self.optimizer_mode.as_ref().and_then(|mode| mode.parse::<char>().ok()),
100            resolc.optimizer_mode
101        );
102        set_if_some!(self.heap_size, resolc.heap_size);
103        set_if_some!(self.stack_size, resolc.stack_size);
104        set_if_some!(
105            self.debug_information.and_then(|v| if v { Some(true) } else { None }),
106            resolc.debug_information
107        );
108
109        resolc
110    }
