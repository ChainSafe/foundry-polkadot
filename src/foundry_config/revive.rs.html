42impl ResolcConfig {
43    /// Returns the `ProjectPathsConfig` sub set of the config.
44    pub fn project_paths(config: &Config) -> ProjectPathsConfig<MultiCompilerLanguage> {
45        let builder = ProjectPathsConfig::builder()
46            .cache(config.cache_path.join(RESOLC_SOLIDITY_FILES_CACHE_FILENAME))
47            .sources(&config.src)
48            .tests(&config.test)
49            .scripts(&config.script)
50            .libs(config.libs.iter())
51            .remappings(config.get_all_remappings())
52            .allowed_path(&config.root)
53            .allowed_paths(&config.libs)
54            .allowed_paths(&config.allow_paths)
55            .include_paths(&config.include_paths)
56            .artifacts(&config.out);
57
58        builder.build_with_root(&config.root)
59    }
60
61    pub fn resolc_settings(config: &Config) -> Result<SolcSettings, SolcError> {

67                config.resolc.debug_information,
