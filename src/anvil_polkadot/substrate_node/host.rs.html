1//! Host functions overrides for Ethereum address recovery.
2//!
3//! The host functions overriding is required because of impersonation feature.
4//! Impersonation is used for sending/executing transactions with a signer
5//! where its private key is unknown. Sent transactions require a signature
6//! that is obtained with sender's private key and the pallet-revive
7//! runtime logic can recover the signer's ethereum address from the signature
8//! and the rest of the transaction bytes. The runtime recovers the signer's
9//! address by running `secp256k1_ecdsa_recover` to recover the signer's public
10//! key, and `keccak_256` to hash the public key to a 20 byes Ethereum address.
11//! These functions are exposed by `sp-io` and used as host functions by the runtime.
12//! This module implements tweaked versions of the host functions from `sp-io`, which
13//! can recognize fake signatures used for impersonated transactions, and can recover
14//! the signer address from them, while expecting those fake signatures to be built in
15//! a certain way ([0; 12] + sender's Ethereum address + [0; 33]).
16//!
17//! The tweaked host functions are especially useful in the context of overriding the
18//! same `sp-io` host functions in the wasm executor type.
19
20use polkadot_sdk::sp_io::{self, EcdsaVerifyError};
21use sp_runtime_interface::{
22    pass_by::{
23        AllocateAndReturnByCodec, AllocateAndReturnPointer, PassFatPointerAndRead,
24        PassPointerAndRead,
25    },
26    runtime_interface,
27};
28
29// The host functions in this module expect transactions
30// with fake signatures conforming the format checked in this function.
31fn is_impersonated(sig: &[u8]) -> bool {
32    sig[..12] == [0; 12] && sig[32..64] == [0; 32]
33}
34
35#[runtime_interface]
36pub trait Crypto {
37    #[version(1)]
38    fn secp256k1_ecdsa_recover(
39        sig: PassPointerAndRead<&[u8; 65], 65>,
40        msg: PassPointerAndRead<&[u8; 32], 32>,
41    ) -> AllocateAndReturnByCodec<Result<[u8; 64], EcdsaVerifyError>> {
42        if is_impersonated(sig) {
43            trace!(
44                target = "host_fn_overrides",
45                name = "secp256k1_ecdsa_recover - version 1",
46                "impersonation for: {:?}",
47                &sig[12..32]
48            );
49            let mut res = [0u8; 64];
50            res[12..32].copy_from_slice(&sig[12..32]);
51            Ok(res)
52        } else {
53            sp_io::crypto::secp256k1_ecdsa_recover(sig, msg)
54        }
55    }
56
57    #[version(2)]
58    fn secp256k1_ecdsa_recover(
59        sig: PassPointerAndRead<&[u8; 65], 65>,
60        msg: PassPointerAndRead<&[u8; 32], 32>,
61    ) -> AllocateAndReturnByCodec<Result<[u8; 64], EcdsaVerifyError>> {
62        if is_impersonated(sig) {
63            trace!(
64                target = "host_fn_overrides",
65                name = "secp256k1_ecdsa_recover - version 2",
66                "impersonation for: {:?}",
67                &sig[12..32]
68            );
69            let mut res = [0u8; 64];
70            res[12..32].copy_from_slice(&sig[12..32]);
71            Ok(res)
72        } else {
73            sp_io::crypto::secp256k1_ecdsa_recover(sig, msg)
74        }
75    }
76}
77
78#[runtime_interface]
79pub trait Hashing {
80    fn keccak_256(data: PassFatPointerAndRead<&[u8]>) -> AllocateAndReturnPointer<[u8; 32], 32> {
81        if data.len() == 64 && is_impersonated(data) {
82            trace!(
83                target = "host_fn_overrides",
84                name = "keccak_256",
85                "impersonation for: {:?}",
86                &data[12..32]
87            );
88            let mut res = [0; 32];
89            res.copy_from_slice(&data[0..32]);
90            res
91        } else {
92            sp_io::hashing::keccak_256(data)
93        }
94    }
95}
