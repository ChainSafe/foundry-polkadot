21#[derive(Debug, thiserror::Error)]
22pub enum BackendError {
23    #[error("Inner client error: {0}")]
24    Client(#[from] sp_blockchain::Error),
25    #[error("Could not find total issuance in the state")]
26    MissingTotalIssuance,
27    #[error("Could not find chain id in the state")]
28    MissingChainId,
29    #[error("Unable to decode total issuance {0}")]
30    DecodeTotalIssuance(codec::Error),
31    #[error("Unable to decode chain id {0}")]
32    DecodeChainId(codec::Error),
33    #[error("Unable to decode balance {0}")]
34    DecodeBalance(codec::Error),
35    #[error("Unable to decode revive account info {0}")]
36    DecodeReviveAccountInfo(codec::Error),
37    #[error("Unable to decode system account info {0}")]
38    DecodeSystemAccountInfo(codec::Error),
39    #[error("Unable to decode revive code info {0}")]
40    DecodeCodeInfo(codec::Error),
41}
42
43type Result<T> = std::result::Result<T, BackendError>;

59    pub fn read_chain_id(&self, hash: Hash) -> Result<u64> {
60        let key = well_known_keys::CHAIN_ID;
61
62        let value = self.read_top_state(hash, key.to_vec())?.ok_or(BackendError::MissingChainId)?;
63        u64::decode(&mut &value[..]).map_err(BackendError::DecodeChainId)
64    }

66    pub fn read_total_issuance(&self, hash: Hash) -> Result<Balance> {
67        let key = well_known_keys::TOTAL_ISSUANCE;
68
69        let value =
70            self.read_top_state(hash, key.to_vec())?.ok_or(BackendError::MissingTotalIssuance)?;
71        Balance::decode(&mut &value[..]).map_err(BackendError::DecodeTotalIssuance)
72    }

74    pub fn read_system_account_info(
75        &self,
76        hash: Hash,

157
158    fn read_top_state(&self, hash: Hash, key: StorageKey) -> Result<Option<StorageValue>> {
159        let maybe_overridden_val = {
160            let mut guard = self.overrides.lock();
161
162            guard.per_block.get(&hash).and_then(|overrides| overrides.top.get(&key).cloned())
163        };
164
165        if let Some(overridden_val) = maybe_overridden_val {
166            return Ok(overridden_val);
167        }
168
169        let state = self.backend.state_at(hash, TrieCacheContext::Trusted)?;
170        Ok(state
171            .storage(key.as_slice())
172            .map_err(|e| sp_blockchain::Error::from_state(Box::new(e)))?)
173    }

176pub type Storage = HashMap<StorageKey, Option<StorageValue>>;
177
178#[derive(Default, Clone)]
179pub struct BlockOverrides {
180    pub top: Storage,
181    pub children: HashMap<StorageKey, Storage>,
182}
183
184pub struct StorageOverrides {
185    // We keep N most recently used block state overrides because we may later get RPC calls which
186    // query the state of past blocks. When state is mutated by the `set_*` RPCs, it gets committed
187    // to the state DB only in the next block.
188    per_block: LruCache<Hash, BlockOverrides>,
189}
190
191impl Default for StorageOverrides {
192    fn default() -> Self {
193        Self { per_block: LruCache::new(NonZeroUsize::new(10).expect("10 is greater than 0")) }
194    }
195}
196
197impl StorageOverrides {
198    pub fn get(&mut self, block: &Hash) -> Option<BlockOverrides> {
199        self.per_block.get(block).cloned()
200    }
201
202    fn set_chain_id(&mut self, latest_block: Hash, id: u64) {
203        let mut changeset = BlockOverrides::default();
204        changeset.top.insert(well_known_keys::CHAIN_ID.to_vec(), Some(id.encode()));
205
206        self.add(latest_block, changeset);
207    }
208
209    #[allow(unused)]
210    fn set_timestamp(&mut self, latest_block: Hash, timestamp: u64) {
211        let mut changeset = BlockOverrides::default();
212        changeset.top.insert(well_known_keys::TIMESTAMP.to_vec(), Some(timestamp.encode()));
213
214        self.add(latest_block, changeset);
215    }
216
217    fn set_system_account_info(
218        &mut self,
219        latest_block: Hash,
220        account_id: AccountId,
221        info: SystemAccountInfo,
222    ) {
223        let mut changeset = BlockOverrides::default();
224        changeset.top.insert(well_known_keys::system_account_info(account_id), Some(info.encode()));
225
226        self.add(latest_block, changeset);
227    }
228
229    fn set_total_issuance(&mut self, latest_block: Hash, value: Balance) {
230        let mut changeset = BlockOverrides::default();
231        changeset.top.insert(well_known_keys::TOTAL_ISSUANCE.to_vec(), Some(value.encode()));
232
233        self.add(latest_block, changeset);
234    }
235
236    fn set_revive_account_info(
237        &mut self,
238        latest_block: Hash,
239        address: Address,
240        info: ReviveAccountInfo,
241    ) {
242        let mut changeset = BlockOverrides::default();
243        changeset.top.insert(
244            well_known_keys::revive_account_info(H160::from_slice(address.as_slice())),
245            Some(info.encode()),
246        );
247
248        self.add(latest_block, changeset);
249    }
250
251    fn set_pristine_code(&mut self, latest_block: Hash, code_hash: H256, code: Option<Bytes>) {
252        let mut changeset = BlockOverrides::default();
253
254        changeset
255            .top
256            .insert(well_known_keys::pristine_code(code_hash), code.map(|code| code.0.encode()));
257
258        self.add(latest_block, changeset);
259    }
260
261    fn set_code_info(&mut self, latest_block: Hash, code_hash: H256, code_info: Option<CodeInfo>) {
262        let mut changeset = BlockOverrides::default();
263
264        changeset.top.insert(
265            well_known_keys::code_info(code_hash),
266            code_info.map(|code_info| code_info.encode()),
267        );
268
269        self.add(latest_block, changeset);
270    }
271
272    fn set_child_storage(
273        &mut self,
274        latest_block: Hash,
275        child_key: StorageKey,
276        key: StorageKey,
277        value: StorageValue,
278    ) {
279        let mut changeset = BlockOverrides::default();
280
281        let mut child_map = Storage::with_capacity(1);
282        child_map.insert(blake2_256(key.as_slice()).to_vec(), Some(value));
283
284        changeset.children.insert(child_key, child_map);
285
286        self.add(latest_block, changeset);
287    }
288
289    fn add(&mut self, latest_block: Hash, changeset: BlockOverrides) {
290        if let Some(per_block) = self.per_block.get_mut(&latest_block) {
291            per_block.top.extend(changeset.top);
292
293            for (child_key, child_map) in changeset.children {
294                per_block.children.entry(child_key).or_default().extend(child_map);
295            }
296        } else {
297            self.per_block.put(latest_block, changeset);
298        }
299    }
