48/// Builds a new service for a full client.
49pub fn new(
50    anvil_config: &AnvilNodeConfig,
51    config: Configuration,
52) -> Result<(Service, TaskManager), ServiceError> {
53    let storage_overrides = Arc::new(Mutex::new(StorageOverrides::default()));
54
55    let (client, backend, keystore, mut task_manager) = client::new_client(
56        anvil_config.get_genesis_number(),
57        &config,
58        sc_service::new_wasm_executor(&config.executor),
59        storage_overrides.clone(),
60    )?;
61
62    let transaction_pool = Arc::from(
63        sc_transaction_pool::Builder::new(
64            task_manager.spawn_essential_handle(),
65            client.clone(),
66            config.role.is_authority().into(),
67        )
68        .with_options(config.transaction_pool.clone())
69        .build(),
70    );
71
72    // Inform the tx pool about imported and finalized blocks.
73    task_manager.spawn_handle().spawn(
74        "txpool-notifications",
75        Some("transaction-pool"),
76        sc_transaction_pool::notification_future(client.clone(), transaction_pool.clone()),
77    );
78
79    let (seal_engine_command_sender, commands_stream) = tokio::sync::mpsc::channel(1024);
80    let commands_stream = ReceiverStream::new(commands_stream);
81
82    let mining_mode =
83        MiningMode::new(anvil_config.block_time, anvil_config.mixed_mining, anvil_config.no_mining);
84    let time_manager = Arc::new(TimeManager::new_with_milliseconds(
85        sp_timestamp::Timestamp::from(
86            anvil_config
87                .get_genesis_timestamp()
88                .checked_mul(1000)
89                .ok_or(ServiceError::Application("Genesis timestamp overflow".into()))?,
90        )
91        .into(),
92    ));
93    let mining_engine = Arc::new(MiningEngine::new(
94        mining_mode,
95        transaction_pool.clone(),
96        time_manager.clone(),
97        seal_engine_command_sender,
98    ));
99
100    let rpc_handlers = spawn_rpc_server(
101        anvil_config.get_genesis_number(),
102        &mut task_manager,
103        client.clone(),
104        config,
105        transaction_pool.clone(),
106        keystore,
107        backend.clone(),
108    )?;
109
110    task_manager.spawn_handle().spawn(
111        "mining_engine_task",
112        Some("consensus"),
113        run_mining_engine(mining_engine.clone()),
114    );
115
116    let proposer = sc_basic_authorship::ProposerFactory::new(
117        task_manager.spawn_handle(),
118        client.clone(),
119        transaction_pool.clone(),
120        None,
121        None,
122    );
123
124    let create_inherent_data_providers = {
125        move |_, ()| {
126            let next_timestamp = time_manager.next_timestamp();
127            async move { Ok(sp_timestamp::InherentDataProvider::new(next_timestamp.into())) }
128        }
129    };
130
131    let params = sc_consensus_manual_seal::ManualSealParams {
132        block_import: client.clone(),
133        env: proposer,
134        client: client.clone(),
135        pool: transaction_pool.clone(),
136        select_chain: SelectChain::new(backend.clone()),
137        commands_stream: Box::pin(commands_stream),
138        consensus_data_provider: None,
139        create_inherent_data_providers,
140    };
141    let authorship_future = sc_consensus_manual_seal::run_manual_seal(params);
142
143    task_manager.spawn_essential_handle().spawn_blocking(
144        "manual-seal",
145        "substrate",
146        authorship_future,
147    );
148
149    Ok((
150        Service {
151            spawn_handle: task_manager.spawn_handle(),
152            client,
153            backend,
154            tx_pool: transaction_pool,
155            rpc_handlers,
156            mining_engine,
157            storage_overrides,
158            genesis_block_number: anvil_config.get_genesis_number(),
159        },
160        task_manager,
161    ))
162}
