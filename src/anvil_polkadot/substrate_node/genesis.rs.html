20/// Genesis settings
21#[derive(Clone, Debug, Default)]
22pub struct GenesisConfig {
23    /// The chain id of the Substrate chain.
24    pub chain_id: u64,
25    /// The initial timestamp for the genesis block in milliseconds
26    pub timestamp: u64,
27    /// All accounts that should be initialised at genesis with their info.
28    pub alloc: Option<BTreeMap<Address, GenesisAccount>>,
29    /// The initial number for the genesis block
30    pub number: u32,
31    /// The genesis header base fee
32    pub base_fee_per_gas: u64,
33    /// The genesis header gas limit.
34    pub gas_limit: Option<u128>,
35}
36
37impl<'a> From<&'a AnvilNodeConfig> for GenesisConfig {
38    fn from(anvil_config: &'a AnvilNodeConfig) -> Self {
39        Self {
40            chain_id: anvil_config.get_chain_id(),
41            // Anvil genesis timestamp is in seconds, while Substrate timestamp is in milliseconds.
42            timestamp: anvil_config
43                .get_genesis_timestamp()
44                .checked_mul(1000)
45                .expect("Genesis timestamp overflow"),
46            alloc: anvil_config.genesis.as_ref().map(|g| g.alloc.clone()),
47            number: anvil_config
48                .get_genesis_number()
49                .try_into()
50                .expect("Genesis block number overflow"),
51            base_fee_per_gas: anvil_config.get_base_fee(),
52            gas_limit: anvil_config.gas_limit,
53        }
54    }
55}

57impl GenesisConfig {
58    pub fn as_storage_key_value(&self) -> Vec<(Vec<u8>, Vec<u8>)> {
59        let storage = vec![
60            (well_known_keys::CHAIN_ID.to_vec(), self.chain_id.encode()),
61            (well_known_keys::TIMESTAMP.to_vec(), self.timestamp.encode()),
62            (well_known_keys::BLOCK_NUMBER_KEY.to_vec(), self.number.encode()),
63        ];
64        // TODO: add other fields
65        storage
66    }
67}

69pub struct DevelopmentGenesisBlockBuilder<Block: BlockT, B, E> {
70    genesis_number: u32,
71    genesis_storage: Storage,
72    commit_genesis_state: bool,
73    backend: Arc<B>,
74    executor: E,
75    _phantom: PhantomData<Block>,
76}
77
78impl<Block: BlockT, B: Backend<Block>, E: RuntimeVersionOf>
79    DevelopmentGenesisBlockBuilder<Block, B, E>
80{
81    pub fn new(
82        genesis_number: u64,
83        build_genesis_storage: &dyn BuildStorage,
84        commit_genesis_state: bool,
85        backend: Arc<B>,
86        executor: E,
87    ) -> sp_blockchain::Result<Self> {
88        let genesis_storage =
89            build_genesis_storage.build_storage().map_err(sp_blockchain::Error::Storage)?;
90        Self::new_with_storage(
91            genesis_number,
92            genesis_storage,
93            commit_genesis_state,
94            backend,
95            executor,
96        )
97    }
98
99    pub fn new_with_storage(
100        genesis_number: u64,
101        genesis_storage: Storage,
102        commit_genesis_state: bool,
103        backend: Arc<B>,
104        executor: E,
105    ) -> sp_blockchain::Result<Self> {
106        Ok(Self {
107            genesis_number: genesis_number.try_into().map_err(|_| {
108                sp_blockchain::Error::Application(
109                    format!(
110                        "Genesis number {} is too large for u32 (max: {})",
111                        genesis_number,
112                        u32::MAX
113                    )
114                    .into(),
115                )
116            })?,
117            genesis_storage,
118            commit_genesis_state,
119            backend,
120            executor,
121            _phantom: PhantomData::<Block>,
122        })
123    }
124}

126impl<Block: BlockT, B: Backend<Block>, E: RuntimeVersionOf> BuildGenesisBlock<Block>
127    for DevelopmentGenesisBlockBuilder<Block, B, E>
128{
129    type BlockImportOperation = <B as Backend<Block>>::BlockImportOperation;
130
131    fn build_genesis_block(self) -> sp_blockchain::Result<(Block, Self::BlockImportOperation)> {
132        let Self {
133            genesis_number,
134            genesis_storage,
135            commit_genesis_state,
136            backend,
137            executor,
138            _phantom,
139        } = self;
140
141        let genesis_state_version =
142            resolve_state_version_from_wasm::<_, HashingFor<Block>>(&genesis_storage, &executor)?;
143        let mut op = backend.begin_operation()?;
144        let state_root =
145            op.set_genesis_state(genesis_storage, commit_genesis_state, genesis_state_version)?;
146        let extrinsics_root = <<<Block as BlockT>::Header as HeaderT>::Hashing as HashT>::trie_root(
147            Vec::new(),
148            genesis_state_version,
149        );
150        let genesis_block = Block::new(
151            <<Block as BlockT>::Header as HeaderT>::new(
152                genesis_number.into(),
153                extrinsics_root,
154                state_root,
155                Default::default(),
156                Default::default(),
157            ),
158            Default::default(),
159        );
160
161        Ok((genesis_block, op))
162    }
163}

165#[cfg(test)]
166mod tests {
167    use super::*;
168
169    #[test]
170    fn test_storage_encoding() {
171        let block_number: u32 = 5;
172        let timestamp: u64 = 10;
173        let chain_id: u64 = 42;
174        let genesis_config =
175            GenesisConfig { number: block_number, timestamp, chain_id, ..Default::default() };
176        let genesis_storage = genesis_config.as_storage_key_value();
177        assert!(
178            genesis_storage
179                .contains(&(well_known_keys::BLOCK_NUMBER_KEY.to_vec(), block_number.encode())),
180            "Block number not found in genesis key-value storage"
181        );
182        assert!(
183            genesis_storage.contains(&(well_known_keys::TIMESTAMP.to_vec(), timestamp.encode())),
184            "Timestamp not found in genesis key-value storage"
185        );
186        assert!(
187            genesis_storage.contains(&(well_known_keys::CHAIN_ID.to_vec(), chain_id.encode())),
188            "Chain id not found in genesis key-value storage"
189        );
190    }
