14/// This is a wrapper around the general Substrate ChainSpec type that allows manual changes to the
15/// genesis block.
16#[derive(Clone)]
17pub struct DevelopmentChainSpec<E = Option<()>, EHF = ()> {
18    inner: GenericChainSpec<E, EHF>,
19    genesis_config: GenesisConfig,
20}
21
22impl<E, EHF> BuildStorage for DevelopmentChainSpec<E, EHF>
23where
24    EHF: HostFunctions,
25    GenericChainSpec<E, EHF>: BuildStorage,
26{
27    fn assimilate_storage(&self, storage: &mut Storage) -> Result<(), String> {
28        self.inner.assimilate_storage(storage)?;
29        storage.top.extend(self.genesis_config.as_storage_key_value());
30        Ok(())
31    }
32}

104fn props() -> Properties {

111pub fn development_chain_spec(
112    genesis_config: GenesisConfig,
113) -> Result<DevelopmentChainSpec, String> {
114    let inner = GenericChainSpec::builder(
115        WASM_BINARY.expect("Development wasm not available"),
116        Default::default(),
117    )
118    .with_name("Development")
119    .with_id("dev")
120    .with_chain_type(ChainType::Development)
121    .with_genesis_config_preset_name(sp_genesis_builder::DEV_RUNTIME_PRESET)
122    .with_properties(props())
123    .build();
124    Ok(DevelopmentChainSpec { inner, genesis_config })
125}
