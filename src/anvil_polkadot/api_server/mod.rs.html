22pub fn spawn(
23    substrate_service: &Service,
24    logging_manager: LoggingManager,
25    config: &AnvilNodeConfig,
26) -> ApiHandle {
27    let (api_handle, receiver) = mpsc::channel(100);
28
29    let service = substrate_service.clone();
30    let mut impersonation_manager = ImpersonationManager::default();
31    impersonation_manager.set_auto_impersonate_account(config.enable_auto_impersonate);
32    substrate_service.spawn_handle.spawn("anvil-api-server", "anvil", async move {
33        let api_server = ApiServer::new(service, receiver, logging_manager, impersonation_manager)
34            .await
35            .unwrap_or_else(|err| panic!("Failed to spawn the API server: {err}"));
36        api_server.run().await;
37    });
38
39    api_handle
40}
