9#[derive(Debug, thiserror::Error)]
10pub enum Error {
11    #[error("Block mining failed: {0}")]
12    Mining(#[from] MiningError),
13    #[error("Rpc Endpoint not implemented")]
14    RpcUnimplemented,
15    #[error("Invalid params: {0}")]
16    InvalidParams(String),
17    #[error("Revive call failed: {0}")]
18    ReviveRpc(#[from] EthRpcError),
19    #[error(transparent)]
20    Backend(#[from] BackendError),
21    #[error("Nonce overflowing the substrate nonce type")]
22    NonceOverflow,
23    #[error(transparent)]
24    RuntimeApi(#[from] sp_api::ApiError),
25    #[error("Error encountered while creating a BalanceWithDust from a U256 balance")]
26    BalanceConversion,
27    #[error("Internal error: {0}")]
28    InternalError(String),
29}

31impl From<subxt::Error> for Error {
32    fn from(err: subxt::Error) -> Self {

37impl From<ClientError> for Error {
38    fn from(err: ClientError) -> Self {

45/// Helper trait to easily convert results to rpc results
46pub(crate) trait ToRpcResponseResult {
47    fn to_rpc_result(self) -> ResponseResult;
48}
49
50/// Converts a serializable value into a `ResponseResult`.
51fn to_rpc_result<T: Serialize>(val: T) -> ResponseResult {
52    match serde_json::to_value(val) {
53        Ok(success) => ResponseResult::Success(success),
54        Err(err) => {
55            error!(%err, "Failed serialize rpc response");
56            ResponseResult::error(RpcError::internal_error())
57        }
58    }
59}
60
61impl<T: Serialize> ToRpcResponseResult for Result<T> {
62    fn to_rpc_result(self) -> ResponseResult {
63        match self {
64            Ok(val) => to_rpc_result(val),
65            Err(err) => match err {
66                Error::Mining(mining_error) => match mining_error {
67                    MiningError::BlockProducing(error) => {
68                        RpcError::internal_error_with(format!("Failed to produce a block: {error}"))
69                            .into()
70                    }
71                    MiningError::MiningModeMismatch => {
72                        RpcError::invalid_params("Current mining mode can not answer this query.")
73                            .into()
74                    }
75                    MiningError::Timestamp => {
76                        RpcError::invalid_params("Current timestamp is newer.").into()
77                    }
78                    MiningError::ClosedChannel => {
79                        RpcError::internal_error_with("Communication channel was dropped.").into()
80                    }
81                },
82                Error::RpcUnimplemented => RpcError::internal_error_with("Not implemented").into(),
83                Error::InvalidParams(error_message) => {
84                    RpcError::invalid_params(error_message).into()
85                }
86                err => RpcError::internal_error_with(format!("{err}")).into(),
87            },
88        }
89    }
90}
