110    pub async fn run(mut self) {
111        while let Some(msg) = self.req_receiver.next().await {
112            let resp = self.execute(msg.req).await;
113
114            msg.resp_sender.send(resp).expect("Dropped receiver");
115        }
116    }

279    async fn mine(&self, blocks: Option<U256>, interval: Option<U256>) -> Result<()> {
280        node_info!("anvil_mine");
281
282        if blocks.is_some_and(|b| u64::try_from(b).is_err()) {
283            return Err(Error::InvalidParams("The number of blocks is too large".to_string()));
284        }
285        if interval.is_some_and(|i| u64::try_from(i).is_err()) {
286            return Err(Error::InvalidParams(
287                "The interval between blocks is too large".to_string(),
288            ));
289        }
290        self.mining_engine
291            .mine(blocks.map(|b| b.to()), interval.map(|i| Duration::from_secs(i.to())))
292            .await
293            .map_err(Error::Mining)
294    }

329    async fn evm_mine_detailed(
330        &self,
331        mine: Option<MineParams<Option<MineOptions>>>,
332    ) -> Result<Vec<Block>> {
333        node_info!("evm_mine_detailed");
334        let mined_blocks = self.mining_engine.do_evm_mine(mine.and_then(|p| p.params)).await?;
335        let mut blocks = Vec::with_capacity(mined_blocks as usize);
336        let last_block = self.client.info().best_number as u64;
337        let starting = last_block - mined_blocks + 1;
338        for block_number in starting..=last_block {
339            if let Some(block) =
340                self.get_block_by_number(BlockNumberOrTag::Number(block_number), true).await?
341            {
342                blocks.push(block);
343            }
344        }
345        Ok(blocks)
346    }

550    async fn send_transaction(
551        &self,
552        transaction_req: WithOtherFields<TransactionRequest>,
553        unsigned_tx: bool,
554    ) -> Result<H256> {
555        node_info!("eth_sendTransaction");
556        let mut transaction = convert_to_generic_transaction(transaction_req.clone().into_inner());
557        let Some(from) = transaction.from else {
558            return Err(Error::ReviveRpc(EthRpcError::InvalidTransaction));
559        };
560
561        let latest_block = self.latest_block();
562        let latest_block_id = Some(BlockId::hash(B256::from_slice(latest_block.as_ref())));
563
564        if transaction.gas.is_none() {
565            transaction.gas =
566                Some(self.estimate_gas(transaction_req.clone(), latest_block_id).await?);
567        }
568
569        if transaction.gas_price.is_none() {
570            transaction.gas_price = Some(self.gas_price().await?);
571        }
572        if transaction.nonce.is_none() {
573            transaction.nonce = Some(self.get_transaction_count(from, latest_block_id).await?);
574        }
575        if transaction.chain_id.is_none() {
576            transaction.chain_id =
577                Some(sp_core::U256::from_big_endian(&self.chain_id(latest_block).to_be_bytes()));
578        }
579
580        let tx = transaction
581            .try_into_unsigned()
582            .map_err(|_| Error::ReviveRpc(EthRpcError::InvalidTransaction))?;
583
584        let payload = if self.impersonation_manager.is_impersonated(from) || unsigned_tx {
585            let mut fake_signature = [0; 65];
586            fake_signature[12..32].copy_from_slice(from.as_bytes());
587            tx.with_signature(fake_signature).signed_payload()
588        } else {
589            let account = self
590                .wallet
591                .accounts
592                .iter()
593                .find(|account| account.address() == from)
594                .ok_or(Error::ReviveRpc(EthRpcError::AccountNotFound(from)))?;

596        };
597
598        self.send_raw_transaction(Bytes(payload)).await
599    }

813    fn set_code(&self, address: Address, bytes: alloy_primitives::Bytes) -> Result<()> {
814        node_info!("anvil_setCode");
815
816        let latest_block = self.latest_block();
817
818        let account_id = self.get_account_id(latest_block, address)?;
819
820        let code_hash = H256(keccak_256(&bytes));
821
822        let maybe_system_account_info =
823            self.backend.read_system_account_info(latest_block, account_id.clone())?;
824        let nonce = maybe_system_account_info.as_ref().map(|info| info.nonce).unwrap_or_default();
825
826        if maybe_system_account_info.is_none() {
827            self.set_frame_system_balance(
828                latest_block,
829                account_id.clone(),
830                substrate_runtime::currency::DOLLARS,
831            )?;
832        }
833
834        let mut old_code_info = None;
835        let revive_account_info = match self
836            .backend
837            .read_revive_account_info(latest_block, address)?
838        {
839            None => {
840                let contract_info = new_contract_info(&address, code_hash, nonce);
841
842                ReviveAccountInfo { account_type: AccountType::Contract(contract_info), dust: 0 }
843            }
844            Some(ReviveAccountInfo { account_type: AccountType::EOA, dust }) => {
845                let contract_info = new_contract_info(&address, code_hash, nonce);
846
847                ReviveAccountInfo { account_type: AccountType::Contract(contract_info), dust }
848            }
849            Some(ReviveAccountInfo {
850                account_type: AccountType::Contract(mut contract_info),
851                dust,
852            }) => {
853                if let Some(code_info) =
854                    self.backend.read_code_info(latest_block, contract_info.code_hash)?
855                {
856                    if code_info.refcount == 1 && contract_info.code_hash != code_hash {
857                        // Remove the pristine code and code info for the old hash.
858                        self.backend.inject_pristine_code(
859                            latest_block,
860                            contract_info.code_hash,
861                            None,
862                        );
863                        self.backend.inject_code_info(latest_block, contract_info.code_hash, None);
864                    }
865
866                    old_code_info = Some(code_info);
867                }
868
869                contract_info.code_hash = code_hash;
870
871                ReviveAccountInfo { account_type: AccountType::Contract(contract_info), dust }
872            }
873        };
874
875        self.backend.inject_revive_account_info(latest_block, address, revive_account_info);
876
877        let code_info = old_code_info
878            .map(|mut code_info| {
879                code_info.code_len = bytes.len() as u32;
880                code_info.code_type = ByteCodeType::Evm;
881                code_info
882            })
883            .unwrap_or_else(|| CodeInfo {
884                owner: <[u8; 32]>::from(account_id).into(),
885                deposit: Default::default(),
886                refcount: 1,
887                code_len: bytes.len() as u32,
888                behaviour_version: 0,
889                code_type: ByteCodeType::Evm,
890            });
891
892        self.backend.inject_pristine_code(latest_block, code_hash, Some(bytes));
893        self.backend.inject_code_info(latest_block, code_hash, Some(code_info));
894
895        Ok(())
896    }

913    fn set_frame_system_balance(
914        &self,
915        latest_block: H256,
916        account_id: AccountId,
917        balance: Balance,
918    ) -> Result<()> {
919        let mut total_issuance = self.backend.read_total_issuance(latest_block)?;
920
921        let mut system_account_info = self
922            .backend
923            .read_system_account_info(latest_block, account_id.clone())?
924            .unwrap_or_else(|| SystemAccountInfo { providers: 1, ..Default::default() });
925
926        if let Some(diff) = balance.checked_sub(system_account_info.data.free) {
927            total_issuance = total_issuance.saturating_add(diff);
928        } else {
929            total_issuance = total_issuance.saturating_sub(system_account_info.data.free - balance);
930        }
931
932        system_account_info.data.free = balance;
933
934        self.backend.inject_system_account_info(latest_block, account_id, system_account_info);
935        self.backend.inject_total_issuance(latest_block, total_issuance);
936
937        Ok(())
938    }

941fn new_contract_info(address: &Address, code_hash: H256, nonce: Nonce) -> ContractInfo {
942    let address = H160::from_slice(address.as_slice());
943
944    let trie_id = {
945        let buf = ("bcontract_trie_v1", address, nonce).using_encoded(BlakeTwo256::hash);
946        buf.as_ref()
947            .to_vec()
948            .try_into()
949            .expect("Runtime uses a reasonable hash size. Hence sizeof(T::Hash) <= 128; qed")
950    };
951
952    ContractInfo {
953        trie_id,
954        code_hash,
955        storage_bytes: 0,
956        storage_items: 0,
957        storage_byte_deposit: 0,
958        storage_item_deposit: 0,
959        storage_base_deposit: 0,
960        immutable_data_len: 0,
961    }
962}

964async fn create_revive_rpc_client(substrate_service: &Service) -> Result<EthRpcClient> {
965    let rpc_client = RpcClient::new(InMemoryRpcClient(substrate_service.rpc_handlers.clone()));
966
967    let genesis_block_number = substrate_service.genesis_block_number.try_into().map_err(|_| {
968        Error::InternalError(format!(
969            "Genesis block number {} is too large for u32 (max: {})",
970            substrate_service.genesis_block_number,
971            u32::MAX
972        ))
973    })?;
974
975    let Some(genesis_hash) = substrate_service.client.hash(genesis_block_number).ok().flatten()
976    else {
977        return Err(Error::InternalError(format!(
978            "Genesis hash not found for genesis block number {}",
979            substrate_service.genesis_block_number
980        )));
981    };
982
983    let Ok(runtime_version) = substrate_service.client.runtime_version_at(genesis_hash) else {
984        return Err(Error::InternalError(
985            "Runtime version not found for given genesis hash".to_string(),
986        ));
987    };
988
989    let subxt_runtime_version = SubxtRuntimeVersion {
990        spec_version: runtime_version.spec_version,
991        transaction_version: runtime_version.transaction_version,
992    };
993
994    let Ok(supported_metadata_versions) =
995        substrate_service.client.runtime_api().metadata_versions(genesis_hash)
996    else {
997        return Err(Error::InternalError("Unable to fetch metadata versions".to_string()));
998    };
999    let Some(latest_metadata_version) = supported_metadata_versions.into_iter().max() else {
1000        return Err(Error::InternalError("No stable metadata versions supported".to_string()));
1001    };
1002    let opaque_metadata = substrate_service
1003        .client
1004        .runtime_api()
1005        .metadata_at_version(genesis_hash, latest_metadata_version)
1006        .map_err(|_| {
1007            Error::InternalError("Failed to get runtime API for genesis hash".to_string())
1008        })?
1009        .ok_or_else(|| {
1010            Error::InternalError(format!(
1011                "Metadata not found for version {latest_metadata_version} at genesis hash"
1012            ))
1013        })?;
1014    let subxt_metadata = SubxtMetadata::decode(&mut (*opaque_metadata).as_slice())
1015        .map_err(|_| Error::InternalError("Unable to decode metadata".to_string()))?;
1016
1017    let api = OnlineClient::<SrcChainConfig>::from_rpc_client_with(
1018        genesis_hash,
1019        subxt_runtime_version,
1020        subxt_metadata,
1021        rpc_client.clone(),
1022    )?;
1023    let rpc = LegacyRpcMethods::<SrcChainConfig>::new(rpc_client.clone());
1024
1025    let block_provider = SubxtBlockInfoProvider::new(api.clone(), rpc.clone()).await?;
1026
1027    let (pool, keep_latest_n_blocks) = {
1028        // see sqlite in-memory issue: https://github.com/launchbadge/sqlx/issues/2510
1029        let pool = SqlitePoolOptions::new()
1030            .max_connections(1)
1031            .idle_timeout(None)
1032            .max_lifetime(None)
1033            .connect("sqlite::memory:")
1034            .await
1035            .map_err(|err| {
1036                Error::ReviveRpc(EthRpcError::ClientError(ClientError::SqlxError(err)))
1037            })?;
1038
1039        (pool, Some(100))
1040    };
1041
1042    let receipt_extractor = ReceiptExtractor::new_with_custom_address_recovery(
1043        api.clone(),
1044        None,
1045        Arc::new(|signed_tx: &TransactionSigned| {
1046            let sig = signed_tx.raw_signature()?;
1047            if sig[..12] == [0; 12] && sig[32..64] == [0; 32] {
1048                let mut res = [0; 20];
1049                res.copy_from_slice(&sig[12..32]);
1050                Ok(H160::from(res))
1051            } else {
1052                signed_tx.recover_eth_address()
1053            }
1054        }),
1055    )
1056    .await
1057    .map_err(|err| Error::ReviveRpc(EthRpcError::ClientError(err)))?;
1058
1059    let receipt_provider = ReceiptProvider::new(
1060        pool,
1061        block_provider.clone(),
1062        receipt_extractor.clone(),
1063        keep_latest_n_blocks,
1064    )
1065    .await
1066    .map_err(|err| Error::ReviveRpc(EthRpcError::ClientError(ClientError::SqlxError(err))))?;
1067
1068    let eth_rpc_client = EthRpcClient::new(api, rpc_client, rpc, block_provider, receipt_provider)
1069        .await
1070        .map_err(Error::from)?;
1071    let eth_rpc_client_clone = eth_rpc_client.clone();
1072    substrate_service.spawn_handle.spawn("block-subscription", "None", async move {
1073        let eth_rpc_client = eth_rpc_client_clone;
1074        let best_future =
1075            eth_rpc_client.subscribe_and_cache_new_blocks(SubscriptionType::BestBlocks);
1076        let finalized_future =
1077            eth_rpc_client.subscribe_and_cache_new_blocks(SubscriptionType::FinalizedBlocks);
1078        let res = tokio::try_join!(best_future, finalized_future).map(|_| ());
1079        if let Err(err) = res {
1080            panic!("Block subscription task failed: {err:?}",)
1081        }
1082    });
1083    Ok(eth_rpc_client)
1084}
