45/// Run the `anvil` command line interface.
46pub fn run() -> Result<()> {
47    setup()?;
48
49    let args = Anvil::parse();
50    args.global.init()?;
51
52    run_command(args)
53}

65/// Run the subcommand.
66pub fn run_command(args: Anvil) -> Result<()> {
67    if let Some(cmd) = &args.cmd {
68        match cmd {
69            AnvilSubcommand::Completions { shell } => {
70                clap_complete::generate(
71                    *shell,
72                    &mut Anvil::command(),
73                    "anvil-polkadot",
74                    &mut std::io::stdout(),
75                );
76            }
77            AnvilSubcommand::GenerateFigSpec => clap_complete::generate(
78                clap_complete_fig::Fig,
79                &mut Anvil::command(),
80                "anvil-polkadot",
81                &mut std::io::stdout(),
82            ),
83        }
84        return Ok(());
85    }
86
87    let (anvil_config, substrate_config) = args.node.into_node_config()?;
88
89    let substrate_client =
90        opts::SubstrateCli { genesis_config: GenesisConfig::from(&anvil_config) };
91
92    let tokio_runtime = build_runtime()?;
93
94    let signals = tokio_runtime.block_on(async { sc_cli::Signals::capture() })?;
95    let config =
96        substrate_client.create_configuration(&substrate_config, tokio_runtime.handle().clone())?;
97    let logging_manager = if anvil_config.enable_tracing {
98        init_tracing(anvil_config.silent)
99    } else {
100        LoggingManager::default()
101    };
102    let runner: sc_cli::Runner<opts::SubstrateCli> =
103        sc_cli::Runner::new(config, tokio_runtime, signals)?;
104
105    Ok(runner.run_node_until_exit(|config| async move {
106        let (_service, task_manager, ..) = spawn(anvil_config, config, logging_manager).await?;
107        Ok::<TaskManager, sc_cli::Error>(task_manager)
108    })?)
109}

111pub async fn spawn(
112    anvil_config: AnvilNodeConfig,
113    substrate_config: sc_service::Configuration,
114    logging_manager: LoggingManager,
115) -> Result<(Service, TaskManager, ApiHandle), sc_cli::Error> {
116    // Spawn the substrate node.
117    let (substrate_service, task_manager) =
118        substrate_node::service::new(&anvil_config, substrate_config)
119            .map_err(sc_cli::Error::Service)?;
120
121    // Spawn the other tasks.
122    let api_handle = spawn_anvil_tasks(anvil_config, &substrate_service, logging_manager)
123        .await
124        .map_err(|err| sc_cli::Error::Application(err.into()))?;
125
126    Ok((substrate_service, task_manager, api_handle))
127}

129pub async fn spawn_anvil_tasks(
130    anvil_config: AnvilNodeConfig,
131    service: &Service,
132    logging_manager: LoggingManager,
133) -> Result<ApiHandle> {
134    // Spawn the api server.
135    let api_handle = api_server::spawn(service, logging_manager, &anvil_config);
136
137    // Spawn the network servers.
138    for addr in &anvil_config.host {
139        let sock_addr = SocketAddr::new(*addr, anvil_config.port);
140
141        // Create a TCP listener.
142        let tcp_listener = tokio::net::TcpListener::bind(sock_addr).await?;
143
144        // Spawn the server future on a new task.
145        let srv =
146            server::serve_on(tcp_listener, anvil_config.server_config.clone(), api_handle.clone());
147        service
148            .spawn_handle
149            .spawn("anvil", "anvil-tcp", async move { srv.await.expect("TCP server failure") });
150    }
151
152    // If configured, spawn the IPC server.
153    anvil_config
154        .get_ipc_path()
155        .map(|path| try_spawn_ipc(&service.spawn_handle, path, api_handle.clone()))
156        .transpose()?;
157
158    anvil_config.print()?;
159
160    Ok(api_handle)
161}

163pub fn init_tracing(silent: bool) -> LoggingManager {
164    use tracing_subscriber::prelude::*;
165
166    let manager = LoggingManager::default();
167    manager.set_enabled(!silent);
168
169    let env_filter = if !silent && std::env::var("RUST_LOG").is_ok() {
170        tracing_subscriber::EnvFilter::from_default_env()
171    } else {
172        tracing_subscriber::EnvFilter::new("warn,node=debug")
173    };
174
175    let _ = if std::env::var("RUST_LOG").is_ok() {
176        tracing_subscriber::Registry::default()
177            .with(env_filter)
178            .with(tracing_subscriber::fmt::layer())
179            .try_init()
180    } else {
181        // Default filter: show substrate warnings/errors and our node targets
182        tracing_subscriber::Registry::default()
183            .with(env_filter)
184            .with(NodeLogLayer::new(manager.clone()))
185            .with(
186                tracing_subscriber::fmt::layer()
187                    .without_time()
188                    .with_target(false)
189                    .with_level(false),
190            )
191            .try_init()
192    };
193
194    manager
195}
