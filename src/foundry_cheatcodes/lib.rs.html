76/// Cheatcode implementation.
77pub(crate) trait Cheatcode: CheatcodeDef + DynCheatcode {
78    /// Applies this cheatcode to the given state.
79    ///
80    /// Implement this function if you don't need access to the EVM data.
81    fn apply(&self, state: &mut Cheatcodes) -> Result {
82        let _ = state;
83        unimplemented!("{}", Self::CHEATCODE.func.id)
84    }
85
86    /// Applies this cheatcode to the given context.
87    ///
88    /// Implement this function if you need access to the EVM data.
89    #[inline(always)]
90    fn apply_stateful(&self, ccx: &mut CheatsCtxt) -> Result {
91        self.apply(ccx.state)
92    }
93
94    /// Applies this cheatcode to the given context and executor.
95    ///
96    /// Implement this function if you need access to the executor.
97    #[inline(always)]
98    fn apply_full(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
99        let _ = executor;
100        self.apply_stateful(ccx)
101    }
102}

104pub trait DynCheatcode: 'static + std::any::Any {
105    fn cheatcode(&self) -> &'static spec::Cheatcode<'static>;
106
107    fn as_debug(&self) -> &dyn std::fmt::Debug;
108
109    fn dyn_apply(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result;
110
111    fn as_any(&self) -> &dyn std::any::Any;
112}
113
114impl<T: Cheatcode> DynCheatcode for T {
115    #[inline]
116    fn cheatcode(&self) -> &'static spec::Cheatcode<'static> {
117        Self::CHEATCODE
118    }
119
120    #[inline]
121    fn as_debug(&self) -> &dyn std::fmt::Debug {
122        self
123    }
124
125    #[inline]
126    fn dyn_apply(&self, ccx: &mut CheatsCtxt, executor: &mut dyn CheatcodesExecutor) -> Result {
127        self.apply_full(ccx, executor)
128    }
129
130    #[inline]
131    fn as_any(&self) -> &dyn std::any::Any {
132        self
133    }
134}

136impl dyn DynCheatcode {
137    pub(crate) fn name(&self) -> &'static str {
138        self.cheatcode().func.signature.split('(').next().unwrap()
139    }
140
141    pub(crate) fn id(&self) -> &'static str {
142        self.cheatcode().func.id
143    }
144
145    pub(crate) fn signature(&self) -> &'static str {
146        self.cheatcode().func.signature
147    }
148
149    pub(crate) fn status(&self) -> &Status<'static> {
150        &self.cheatcode().status
151    }
152}

154/// The cheatcode context, used in `Cheatcode`.
155pub struct CheatsCtxt<'cheats, 'evm, 'db, 'db2> {
156    /// The cheatcodes inspector state.
157    pub state: &'cheats mut Cheatcodes,
158    /// The EVM data.
159    pub ecx: &'evm mut EthEvmContext<&'db mut (dyn DatabaseExt + 'db2)>,
160    /// The original `msg.sender`.
161    pub(crate) caller: Address,
162    /// Gas limit of the current cheatcode call.
163    pub(crate) gas_limit: u64,
164}
165
166impl<'db, 'db2> std::ops::Deref for CheatsCtxt<'_, '_, 'db, 'db2> {
167    type Target = EthEvmContext<&'db mut (dyn DatabaseExt + 'db2)>;
168
169    #[inline(always)]
170    fn deref(&self) -> &Self::Target {
171        self.ecx
172    }
173}
174
175impl std::ops::DerefMut for CheatsCtxt<'_, '_, '_, '_> {
176    #[inline(always)]
177    fn deref_mut(&mut self) -> &mut Self::Target {
178        &mut *self.ecx
179    }
180}
181
182impl CheatsCtxt<'_, '_, '_, '_> {
183    #[inline]
184    pub(crate) fn is_precompile(&self, address: &Address) -> bool {
185        self.ecx.journaled_state.inner.precompiles.contains(address)
186    }
