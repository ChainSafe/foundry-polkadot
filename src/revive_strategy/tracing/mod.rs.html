16// Traces the execution inside pallet_revive.
17// This is a temporary solution to the fact that custom Tracer is not implementable for the time
18// being.
19pub fn trace<T: Config, R, F: FnOnce() -> R>(f: F) -> (R, Option<CallTrace<U256>>, PrestateTrace) {
20    let mut call_tracer = CallTracer::new(
21        Default::default(),
22        Pallet::<revive_env::Runtime>::evm_gas_from_weight as fn(Weight) -> U256,
23    );
24
25    let mut prestate_tracer: PrestateTracer<revive_env::Runtime> =
26        PrestateTracer::new(PrestateTracerConfig {
27            diff_mode: true,
28            disable_storage: false,
29            disable_code: false,
30        });
31
32    let result = trace_revive(&mut prestate_tracer, || trace_revive(&mut call_tracer, f));
33    let prestate_trace = prestate_tracer.collect_trace();
34    let calls = call_tracer.collect_trace();
35    (result, calls, prestate_trace)
36}

38/// Applies `PrestateTrace` diffs to the revm state
39pub fn apply_prestate_trace(prestate_trace: PrestateTrace, ecx: Ecx<'_, '_, '_>) {
40    match prestate_trace {
41        polkadot_sdk::pallet_revive::evm::PrestateTrace::DiffMode { pre: _, post } => {
42            for (key, PrestateTraceInfo { balance, nonce, code, storage }) in post {
43                let address = Address::from_slice(key.as_bytes());
44                let account = ecx
45                    .journaled_state
46                    .load_account(address)
47                    .expect("account could not be loaded")
48                    .data;
49
50                account.mark_touch();
51
52                if let Some(balance) = balance {
53                    account.info.balance = RU256::from_limbs(balance.0);
54                };
55
56                if let Some(nonce) = nonce {
57                    account.info.nonce = nonce.into();
58                };
59
60                if let Some(code) = code {
61                    let account =
62                        ecx.journaled_state.state.get_mut(&address).expect("account is loaded");
63                    let bytecode = Bytecode::new_raw(Bytes::from(code.0));
64                    account.info.code_hash = bytecode.hash_slow();
65                    account.info.code = Some(bytecode);
66                }
67                ecx.journaled_state.load_account(address).expect("account could not be loaded");
68
69                ecx.journaled_state.touch(address);
70                for (slot, entry) in storage {
71                    let key = RU256::from_be_slice(&slot.0);
72                    if let Some(e_entry) = entry {
73                        let entry = RU256::from_be_slice(&e_entry.0);
74
75                        ecx.journaled_state.sstore(address, key, entry).expect("to succeed");
76                    }
77                }
78            }
79        }
80        _ => panic!("Can't happen"),
81    };
