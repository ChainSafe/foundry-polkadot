38impl ExecutorStrategyRunner for ReviveExecutorStrategyRunner {

51    /// Sets the balance of an account.
52    ///
53    /// Amount should be in the range of [0, u128::MAX] despite the type
54    /// because Ethereum balances are u256 while Polkadot balances are u128.
55    fn set_balance(
56        &self,
57        executor: &mut foundry_evm::executors::Executor,
58        address: Address,
59        amount: U256,
60    ) -> foundry_evm::backend::BackendResult<()> {
61        let amount_pvm =
62            sp_core::U256::from_little_endian(&amount.as_le_bytes()).min(u128::MAX.into());
63        let balance_native =
64            BalanceWithDust::<BalanceOf<Runtime>>::from_value::<Runtime>(amount_pvm).unwrap();
65
66        EvmExecutorStrategyRunner.set_balance(executor, address, amount)?;
67
68        let min_balance = pallet_balances::Pallet::<Runtime>::minimum_balance();
69
70        execute_with_externalities(|externalities| {
71            externalities.execute_with(|| {
72                pallet_balances::Pallet::<Runtime>::set_balance(
73                    &AccountId::to_fallback_account_id(&H160::from_slice(address.as_slice())),
74                    balance_native.into_rounded_balance().saturating_add(min_balance),
75                );
76            })
77        });
78        Ok(())
79    }
80
81    fn get_balance(
82        &self,
83        executor: &foundry_evm::executors::Executor,
84        address: Address,
85    ) -> foundry_evm::backend::BackendResult<U256> {
86        let evm_balance = EvmExecutorStrategyRunner.get_balance(executor, address)?;
87
88        Ok(evm_balance)
89    }
90
91    fn set_nonce(
92        &self,
93        executor: &mut foundry_evm::executors::Executor,
94        address: Address,
95        nonce: u64,
96    ) -> foundry_evm::backend::BackendResult<()> {
97        EvmExecutorStrategyRunner.set_nonce(executor, address, nonce)?;
98        execute_with_externalities(|externalities| {
99            externalities.execute_with(|| {
100                let account_id =
101                    AccountId::to_fallback_account_id(&H160::from_slice(address.as_slice()));
102                let current_nonce = System::account_nonce(&account_id);
103
104                assert!(
105                    current_nonce as u64 <= nonce,
106                    "Cannot set nonce lower than current nonce: {current_nonce} > {nonce}"
107                );
108
109                while (System::account_nonce(&account_id) as u64) < nonce {
110                    System::inc_account_nonce(&account_id);
111                }
112            })
113        });
114        Ok(())
115    }
116
117    fn get_nonce(
118        &self,
119        executor: &foundry_evm::executors::Executor,
120        address: Address,
121    ) -> foundry_evm::backend::BackendResult<u64> {
122        let evm_nonce = EvmExecutorStrategyRunner.get_nonce(executor, address)?;
123        let revive_nonce = execute_with_externalities(|externalities| {
124            externalities.execute_with(|| {
125                System::account_nonce(AccountId::to_fallback_account_id(&H160::from_slice(
126                    address.as_slice(),
127                )))
128            })
129        });
130
131        assert_eq!(evm_nonce, revive_nonce as u64);
132        Ok(evm_nonce)
133    }

168impl ExecutorStrategyExt for ReviveExecutorStrategyRunner {
169    fn revive_set_dual_compiled_contracts(
170        &self,
171        ctx: &mut dyn ExecutorStrategyContext,
172        dual_compiled_contracts: DualCompiledContracts,
173    ) {
174        let ctx = get_context_ref_mut(ctx);
175        ctx.dual_compiled_contracts = dual_compiled_contracts;
176    }
177
178    fn revive_set_compilation_output(
179        &self,
180        ctx: &mut dyn ExecutorStrategyContext,
181        output: ProjectCompileOutput,
182    ) {
183        let ctx = get_context_ref_mut(ctx);
184        ctx.compilation_output.replace(output);
185    }
