62/// Controls the automatic migration to PVM mode during test execution.
63#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
64pub enum PvmStartupMigration {
65    /// Defer database migration to a later execution point.
66    /// This is the initial state - waiting for the test contract to be deployed.
67    Defer,
68    /// Allow database migration to PVM.
69    /// Set by `base_contract_deployed()` when the test contract is deployed.
70    #[default]
71    Allow,
72    /// Database migration has already been performed.
73    /// Prevents redundant migrations.
74    Done,
75}
76
77impl PvmStartupMigration {
78    /// Check if startup migration is allowed
79    pub fn is_allowed(&self) -> bool {
80        matches!(self, Self::Allow)
81    }
82
83    /// Allow migrating the database to PVM storage
84    pub fn allow(&mut self) {
85        *self = Self::Allow;
86    }
87
88    /// Mark the migration as completed
89    pub fn done(&mut self) {
90        *self = Self::Done;
91    }
92}

131fn set_nonce(address: Address, nonce: u64, ecx: Ecx<'_, '_, '_>) {
132    execute_with_externalities(|externalities| {
133        externalities.execute_with(|| {
134            let account_id =
135                AccountId::to_fallback_account_id(&H160::from_slice(address.as_slice()));
136            let current_nonce = System::account_nonce(&account_id);
137
138            assert!(
139                current_nonce as u64 <= nonce,
140                "Cannot set nonce lower than current nonce: {current_nonce} > {nonce}"
141            );
142
143            while (System::account_nonce(&account_id) as u64) < nonce {
144                System::inc_account_nonce(&account_id);
145            }
146        })
147    });
148    let account = ecx.journaled_state.load_account(address).expect("account loaded").data;
149    account.mark_touch();
150    account.info.nonce = nonce;
151}

153fn set_balance(address: Address, amount: U256, ecx: Ecx<'_, '_, '_>) -> U256 {
154    let account = ecx.journaled_state.load_account(address).expect("account loaded").data;
155    account.mark_touch();
156    account.info.balance = amount;
157    let amount_pvm = sp_core::U256::from_little_endian(&amount.as_le_bytes()).min(u128::MAX.into());
158    let balance_native =
159        BalanceWithDust::<BalanceOf<Runtime>>::from_value::<Runtime>(amount_pvm).unwrap();
160
161    let min_balance = pallet_balances::Pallet::<Runtime>::minimum_balance();
162
163    let old_balance = execute_with_externalities(|externalities| {
164        externalities.execute_with(|| {
165            let addr = &AccountId::to_fallback_account_id(&H160::from_slice(address.as_slice()));
166            let old_balance = pallet_revive::Pallet::<Runtime>::evm_balance(&H160::from_slice(
167                address.as_slice(),
168            ));
169            pallet_balances::Pallet::<Runtime>::set_balance(
170                addr,
171                balance_native.into_rounded_balance().saturating_add(min_balance),
172            );
173            old_balance
174        })
175    });
176    U256::from_limbs(old_balance.0)
177}

207impl CheatcodeInspectorStrategyRunner for PvmCheatcodeInspectorStrategyRunner {
208    fn apply_full(
209        &self,
210        cheatcode: &dyn foundry_cheatcodes::DynCheatcode,
211        ccx: &mut CheatsCtxt<'_, '_, '_, '_>,
212        executor: &mut dyn foundry_cheatcodes::CheatcodesExecutor,
213    ) -> Result {
214        fn is<T: std::any::Any>(t: TypeId) -> bool {
215            TypeId::of::<T>() == t
216        }
217        let using_pvm = get_context_ref_mut(ccx.state.strategy.context.as_mut()).using_pvm;
218
219        match cheatcode.as_any().type_id() {
220            t if is::<pvmCall>(t) => {
221                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
222                let pvmCall { enabled } = cheatcode.as_any().downcast_ref().unwrap();
223                let ctx: &mut PvmCheatcodeInspectorStrategyContext =
224                    get_context_ref_mut(ccx.state.strategy.context.as_mut());
225                if *enabled {
226                    select_pvm(ctx, ccx.ecx);
227                } else {
228                    select_evm(ctx, ccx.ecx);
229                }
230                Ok(Default::default())
231            }
232            t if using_pvm && is::<dealCall>(t) => {
233                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
234
235                let &dealCall { account, newBalance } = cheatcode.as_any().downcast_ref().unwrap();
236
237                let old_balance = set_balance(account, newBalance, ccx.ecx);
238                let record = DealRecord { address: account, old_balance, new_balance: newBalance };
239                ccx.state.eth_deals.push(record);
240                Ok(Default::default())
241            }
242            t if using_pvm && is::<setNonceCall>(t) => {
243                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
244
245                let &setNonceCall { account, newNonce } =
246                    cheatcode.as_any().downcast_ref().unwrap();
247                set_nonce(account, newNonce, ccx.ecx);
248
249                Ok(Default::default())
250            }
251            t if using_pvm && is::<setNonceUnsafeCall>(t) => {
252                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
253                // TODO implement unsafe_set_nonce on polkadot-sdk
254                let &setNonceUnsafeCall { account, newNonce } =
255                    cheatcode.as_any().downcast_ref().unwrap();
256                set_nonce(account, newNonce, ccx.ecx);
257                Ok(Default::default())
258            }
259            t if using_pvm && is::<getNonce_0Call>(t) => {
260                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
261                let &getNonce_0Call { account } = cheatcode.as_any().downcast_ref().unwrap();
262                let nonce = execute_with_externalities(|externalities| {
263                    externalities.execute_with(|| {
264                        System::account_nonce(AccountId::to_fallback_account_id(&H160::from_slice(
265                            account.as_slice(),
266                        )))
267                    })
268                });
269                Ok(u64::from(nonce).abi_encode())
270            }
271            t if using_pvm && is::<rollCall>(t) => {
272                let &rollCall { newHeight } = cheatcode.as_any().downcast_ref().unwrap();
273
274                set_block_number(newHeight, ccx.ecx);
275
276                Ok(Default::default())
277            }
278            t if using_pvm && is::<warpCall>(t) => {
279                let &warpCall { newTimestamp } = cheatcode.as_any().downcast_ref().unwrap();
280
281                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
282                set_timestamp(newTimestamp, ccx.ecx);
283
284                Ok(Default::default())
285            }
286            t if using_pvm && is::<loadCall>(t) => {
287                tracing::info!(cheatcode = ?cheatcode.as_debug() , using_pvm = ?using_pvm);
288                let &loadCall { target, slot } = cheatcode.as_any().downcast_ref().unwrap();
289                let target_address_h160 = H160::from_slice(target.as_slice());
290                let storage_value = execute_with_externalities(|externalities| {
291                    externalities.execute_with(|| {
292                        Pallet::<Runtime>::get_storage(target_address_h160, slot.into())
293                    })
294                });
295                let result = storage_value
296                    .ok()
297                    .flatten()
298                    .map(|b| B256::from_slice(&b))
299                    .unwrap_or(B256::ZERO);
300                Ok(result.abi_encode())
301            }
302            // Not custom, just invoke the default behavior
303            _ => cheatcode.dyn_apply(ccx, executor),
304        }
305    }

376    fn pre_step_end(
377        &self,
378        ctx: &mut dyn CheatcodeInspectorStrategyContext,
379        interpreter: &mut Interpreter,
380        _ecx: Ecx<'_, '_, '_>,
381    ) -> bool {
382        let ctx = get_context_ref_mut(ctx);
383
384        if !ctx.using_pvm {
385            return false;
386        }
387
388        let address = match interpreter.bytecode.opcode() {
389            op::SELFBALANCE => interpreter.input.target_address,
390            op::BALANCE => {
391                if interpreter.stack.is_empty() {
392                    return true;
393                }
394
395                Address::from_word(B256::from(unsafe { interpreter.stack.pop_unsafe() }))
396            }
397            _ => return true,
398        };
399
400        let balance = execute_with_externalities(|externalities| {
401            externalities.execute_with(|| {
402                Pallet::<Runtime>::evm_balance(&H160::from_slice(address.as_slice()))
403            })
404        });
405        let balance = U256::from_limbs(balance.0);
406        tracing::info!(operation = "get_balance" , using_pvm = ?ctx.using_pvm, target = ?address, balance = ?balance);
407
408        // Skip the current BALANCE instruction since we've already handled it
409        if interpreter.stack.push(balance) {
410            interpreter.bytecode.relative_jump(1);
411        } else {
412            // stack overflow; nothing else to do here
413        }
414
415        false // Let EVM handle all operations
416    }
417}

419fn select_pvm(ctx: &mut PvmCheatcodeInspectorStrategyContext, data: Ecx<'_, '_, '_>) {
420    if ctx.using_pvm {
421        tracing::info!("already in PVM");
422        return;
423    }
424
425    tracing::info!("switching to PVM");
426    ctx.using_pvm = true;
427
428    let block_number = data.block.number;
429    let timestamp = data.block.timestamp;
430
431    execute_with_externalities(|externalities| {
432        externalities.execute_with(|| {
433            System::set_block_number(block_number.saturating_to());
434            Timestamp::set_timestamp(timestamp.saturating_to::<u64>() * 1000);
435
436            let test_contract = data.journaled_state.database.get_test_contract_address();
437            let persistent_accounts = data.journaled_state.database.persistent_accounts().clone();
438
439            for address in persistent_accounts.into_iter().chain([data.tx.caller]) {
440                let acc = data.journaled_state.load_account(address).expect("failed to load account");
441                let amount = acc.data.info.balance;
442                let nonce = acc.data.info.nonce;
443                let account_id =
444                    AccountId::to_fallback_account_id(&H160::from_slice(address.as_slice()));
445
446                // Convert EVM balance to PVM balance with precision handling
447                // TODO: needs to be replaced with `set_evm_balance`` once new pallet-revive is used
448                let amount_pvm =
449                    sp_core::U256::from_little_endian(&amount.as_le_bytes()).min(u128::MAX.into());
450                let balance_native =
451                    BalanceWithDust::<BalanceOf<Runtime>>::from_value::<Runtime>(amount_pvm).unwrap();
452                let balance = Pallet::<Runtime>::convert_native_to_evm(balance_native);
453                let amount_evm = U256::from_limbs(balance.0);
454
455                if amount != amount_evm {
456                    let _ = sh_err!(
457                        "Amount mismatch {amount} != {amount_evm}, Polkadot balances are u128. Test results may be incorrect."
458                    );
459                }
460
461                let min_balance = pallet_balances::Pallet::<Runtime>::minimum_balance();
462                <Runtime as Config>::Currency::set_balance(
463                    &account_id,
464                    balance_native.into_rounded_balance().saturating_add(min_balance),
465                );
466                // END OF THE BLOCK TO BE REMOVED
467
468                let current_nonce = System::account_nonce(&account_id);
469                assert!(
470                    current_nonce as u64 <= nonce,
471                    "Cannot set nonce lower than current nonce: {current_nonce} > {nonce}"
472                );
473
474                while (System::account_nonce(&account_id) as u64) < nonce {
475                    System::inc_account_nonce(&account_id);
476                }
477
478                // TODO handle immutables
479                // Migrate bytecode for deployed contracts (skip test contract)
480                if test_contract != Some(address)
481                    && let Some(bytecode) = acc.data.info.code.as_ref() {
482
483                    let account_h160 = H160::from_slice(address.as_slice());
484
485                    // Skip if contract already exists in PVM
486                    if AccountInfo::<Runtime>::load_contract(&account_h160).is_none() {
487                        if let Some(pvm_bytecode) = ctx.dual_compiled_contracts
488                            .find_by_evm_deployed_bytecode_with_immutables(bytecode.original_byte_slice())
489                            .and_then(|(_, contract)| {
490                                contract.resolc_bytecode.as_bytes()
491                            })
492                        {
493                            let origin = OriginFor::<Runtime>::signed(Pallet::<Runtime>::account_id());
494                            let code_hash = Pallet::<Runtime>::bare_upload_code(
495                                origin,
496                                pvm_bytecode.to_vec(),
497                                BalanceOf::<Runtime>::MAX,
498                            )
499                            .ok()
500                            .map(|upload_result| upload_result.code_hash)
501                            .expect("Failed to upload PVM bytecode");
502
503                            let contract_info = ContractInfo::<Runtime>::new(&account_h160, nonce as u32, code_hash)
504                                .expect("Failed to create contract info");
505
506                            AccountInfo::<Runtime>::insert_contract(&account_h160, contract_info);
507
508                        } else {
509                            tracing::info!(
510                                address = ?address,
511                                "no PVM equivalent found for EVM bytecode, skipping migration"
512                            );
513                        }
514                    }
515                }
516            }
517        })
518    });
519}

521fn select_evm(ctx: &mut PvmCheatcodeInspectorStrategyContext, data: Ecx<'_, '_, '_>) {
522    if !ctx.using_pvm {
523        tracing::info!("already in EVM");
524        return;
525    }
526
527    tracing::info!("switching to EVM");
528    ctx.using_pvm = false;
529
530    execute_with_externalities(|externalities| {
531        externalities.execute_with(|| {
532            let block_number = System::block_number();
533            let timestamp = Timestamp::get();
534
535            data.block.number = U256::from(block_number);
536            data.block.timestamp = U256::from(timestamp / 1000);
537
538            let test_contract = data.journaled_state.database.get_test_contract_address();
539            let persistent_accounts = data.journaled_state.database.persistent_accounts().clone();
540            for address in persistent_accounts.into_iter().chain([data.tx.caller]) {
541                let account_evm = H160::from_slice(address.as_slice());
542                let pallet_evm_nonce = Pallet::<Runtime>::evm_nonce(&account_evm);
543                let pallet_evm_balance = Pallet::<Runtime>::evm_balance(&account_evm);
544                let amount_evm = U256::from_limbs(pallet_evm_balance.0);
545                let account = journaled_account(data, address).expect("failed to load account");
546                account.info.balance = amount_evm;
547                account.info.nonce = pallet_evm_nonce as u64;
548
549                // Migrate bytecode for deployed contracts (skip test contract)
550                if test_contract != Some(address)
551                    && let Some(info) = AccountInfo::<Runtime>::load_contract(&account_evm)
552                {
553                    let hash = hex::encode(info.code_hash);
554                    if let Some((code_hash, bytecode)) = ctx
555                        .dual_compiled_contracts
556                        .find_by_resolc_bytecode_hash(hash)
557                        .and_then(|(_, contract)| {
558                            contract.evm_deployed_bytecode.as_bytes().map(|evm_bytecode| {
559                                (
560                                    contract.evm_bytecode_hash,
561                                    Bytecode::new_raw(evm_bytecode.clone()),
562                                )
563                            })
564                        })
565                    {
566                        account.info.code_hash = code_hash;
567                        account.info.code = Some(bytecode);
568                    } else {
569                        tracing::info!(
570                            address = ?address,
571                            "no EVM equivalent found for PVM bytecode, skipping migration"
572                        );
573                    }
574                }
575            }
576        })
577    });

580impl foundry_cheatcodes::CheatcodeInspectorStrategyExt for PvmCheatcodeInspectorStrategyRunner {
581    /// Try handling the `CREATE` within PVM.
582    ///
583    /// If `Some` is returned then the result must be returned immediately, else the call must be
584    /// handled in EVM.
585    fn revive_try_create(
586        &self,
587        state: &mut foundry_cheatcodes::Cheatcodes,
588        ecx: Ecx<'_, '_, '_>,
589        input: &dyn CommonCreateInput,
590        _executor: &mut dyn foundry_cheatcodes::CheatcodesExecutor,
591    ) -> Option<CreateOutcome> {
592        let ctx = get_context_ref_mut(state.strategy.context.as_mut());
593
594        if !ctx.using_pvm {
595            return None;
596        }
597
598        if let Some(CreateScheme::Create) = input.scheme() {
599            let caller = input.caller();
600            let nonce = ecx
601                .journaled_state
602                .load_account(input.caller())
603                .expect("to load caller account")
604                .info
605                .nonce;
606            let address = caller.create(nonce);
607            if ecx
608                .journaled_state
609                .database
610                .get_test_contract_address()
611                .map(|addr| address == addr)
612                .unwrap_or_default()
613            {
614                tracing::info!(
615                    "running create in EVM, instead of PVM (Test Contract) {:#?}",
616                    address
617                );
618                return None;
619            }
620        }
621
622        let init_code = input.init_code();
623        tracing::info!("running create in PVM");
624
625        let find_contract = ctx
626            .dual_compiled_contracts
627            .find_bytecode(&init_code.0)
628            .unwrap_or_else(|| panic!("failed finding contract for {init_code:?}"));
629
630        let constructor_args = find_contract.constructor_args();
631        let contract = find_contract.contract();
632
633        let max_gas =
634            <<Runtime as Config>::EthGasEncoder as GasEncoder<BalanceOf<Runtime>>>::encode(
635                Default::default(),
636                Weight::MAX,
637                1u128 << 99,
638            );
639        let gas_limit = sp_core::U256::from(input.gas_limit()).min(max_gas);
640
641        let (res, _call_trace, prestate_trace) = execute_with_externalities(|externalities| {
642            externalities.execute_with(|| {
643                trace::<Runtime, _, _>(|| {
644                    // TODO: Find a way how to do it correctly
645                    // Use pallet-revive origin to bypass EIP-3607.
646                    let origin = OriginFor::<Runtime>::signed(Pallet::<Runtime>::account_id());
647                    let evm_value = sp_core::U256::from_little_endian(&input.value().as_le_bytes());
648
649                    let (gas_limit, storage_deposit_limit) =
650                    <<Runtime as Config>::EthGasEncoder as GasEncoder<BalanceOf<Runtime>>>::decode(
651                        gas_limit,
652                    )
653                    .expect("gas limit is valid");
654                    let storage_deposit_limit = DepositLimit::Balance(storage_deposit_limit);
655                    let code = Code::Upload(contract.resolc_bytecode.as_bytes().unwrap().to_vec());
656                    let data = constructor_args.to_vec();
657                    let salt = match input.scheme() {
658                        Some(CreateScheme::Create2 { salt }) => Some(
659                            salt.as_limbs()
660                                .iter()
661                                .flat_map(|&x| x.to_le_bytes())
662                                .collect::<Vec<u8>>()
663                                .try_into()
664                                .unwrap(),
665                        ),
666                        _ => None,
667                    };
668                    let bump_nonce = BumpNonce::Yes;
669
670                    Pallet::<Runtime>::bare_instantiate(
671                        origin,
672                        evm_value,
673                        gas_limit,
674                        storage_deposit_limit,
675                        code,
676                        data,
677                        salt,
678                        bump_nonce,
679                    )
680                })
681            })
682        });
683
684        let mut gas = Gas::new(input.gas_limit());
685        let gas_used =
686            <<Runtime as Config>::EthGasEncoder as GasEncoder<BalanceOf<Runtime>>>::encode(
687                gas_limit,
688                res.gas_required,
689                res.storage_deposit.charge_or_zero(),
690            );
691        let result = match &res.result {
692            Ok(result) => {
693                let _ = gas.record_cost(gas_used.as_u64());
694
695                let outcome = if result.result.did_revert() {
696                    CreateOutcome {
697                        result: InterpreterResult {
698                            result: InstructionResult::Revert,
699                            output: result.result.data.clone().into(),
700                            gas,
701                        },
702                        address: None,
703                    }
704                } else {
705                    CreateOutcome {
706                        result: InterpreterResult {
707                            result: InstructionResult::Return,
708                            output: contract.resolc_bytecode.as_bytes().unwrap().clone(),
709                            gas,
710                        },
711                        address: Some(Address::from_slice(result.addr.as_bytes())),
712                    }
713                };
714
715                Some(outcome)
716            }
717            Err(e) => {
718                tracing::error!("Contract creation failed: {e:#?}");
719                Some(CreateOutcome {
720                    result: InterpreterResult {
721                        result: InstructionResult::Revert,
722                        output: Bytes::from_iter(
723                            format!("Contract creation failed: {e:#?}").as_bytes(),
724                        ),
725                        gas,
726                    },
727                    address: None,
728                })
729            }
730        };
731
732        apply_prestate_trace(prestate_trace, ecx);
733
734        result
735    }
736
737    /// Try handling the `CALL` within PVM.
738    ///
739    /// If `Some` is returned then the result must be returned immediately, else the call must be
740    /// handled in EVM.
741    fn revive_try_call(
742        &self,
743        state: &mut foundry_cheatcodes::Cheatcodes,
744        ecx: Ecx<'_, '_, '_>,
745        call: &CallInputs,
746        _executor: &mut dyn foundry_cheatcodes::CheatcodesExecutor,
747    ) -> Option<CallOutcome> {
748        let ctx = get_context_ref_mut(state.strategy.context.as_mut());
749
750        if !ctx.using_pvm {
751            return None;
752        }
753
754        if ecx
755            .journaled_state
756            .database
757            .get_test_contract_address()
758            .map(|addr| call.bytecode_address == addr)
759            .unwrap_or_default()
760        {
761            tracing::info!(
762                "running call in EVM, instead of PVM (Test Contract) {:#?}",
763                call.bytecode_address
764            );
765            return None;
766        }
767
768        tracing::info!("running call in PVM {:#?}", call);
769
770        let max_gas =
771            <<Runtime as Config>::EthGasEncoder as GasEncoder<BalanceOf<Runtime>>>::encode(
772                Default::default(),
773                Weight::MAX,
774                1u128 << 99,
775            );
776        let gas_limit = sp_core::U256::from(call.gas_limit).min(max_gas);
777
778        let (res, _call_trace, prestate_trace) = execute_with_externalities(|externalities| {
779            externalities.execute_with(|| {
780                trace::<Runtime, _, _>(|| {
781                    let origin = OriginFor::<Runtime>::signed(AccountId::to_fallback_account_id(
782                        &H160::from_slice(call.caller.as_slice()),
783                    ));
784                    let evm_value =
785                        sp_core::U256::from_little_endian(&call.call_value().as_le_bytes());
786
787                    let (gas_limit, storage_deposit_limit) =
788                    <<Runtime as Config>::EthGasEncoder as GasEncoder<BalanceOf<Runtime>>>::decode(
789                        gas_limit,
790                    )
791                    .expect("gas limit is valid");
792                    let storage_deposit_limit = DepositLimit::Balance(storage_deposit_limit);
793                    let target = H160::from_slice(call.target_address.as_slice());
794
795                    Pallet::<Runtime>::bare_call(
796                        origin,
797                        target,
798                        evm_value,
799                        gas_limit,
800                        storage_deposit_limit,
801                        call.input.bytes(ecx).to_vec(),
802                    )
803                })
804            })
805        });
806
807        let mut gas = Gas::new(call.gas_limit);
808        let gas_used =
809            <<Runtime as Config>::EthGasEncoder as GasEncoder<BalanceOf<Runtime>>>::encode(
810                gas_limit,
811                res.gas_required,
812                res.storage_deposit.charge_or_zero(),
813            );
814        let result = match res.result {
815            Ok(result) => {
816                let _ = gas.record_cost(gas_used.as_u64());
817                let outcome = if result.did_revert() {
818                    tracing::error!("Contract call reverted");
819                    CallOutcome {
820                        result: InterpreterResult {
821                            result: InstructionResult::Revert,
822                            output: result.data.into(),
823                            gas,
824                        },
825                        memory_offset: call.return_memory_offset.clone(),
826                    }
827                } else {
828                    CallOutcome {
829                        result: InterpreterResult {
830                            result: InstructionResult::Return,
831                            output: result.data.into(),
832                            gas,
833                        },
834                        memory_offset: call.return_memory_offset.clone(),
835                    }
836                };
837
838                Some(outcome)
839            }
840            Err(e) => {
841                tracing::error!("Contract call failed: {e:#?}");
842                Some(CallOutcome {
843                    result: InterpreterResult {
844                        result: InstructionResult::Revert,
845                        output: Bytes::from_iter(
846                            format!("Contract call failed: {e:#?}").as_bytes(),
847                        ),
848                        gas,
849                    },
850                    memory_offset: call.return_memory_offset.clone(),
851                })
852            }
853        };
854
855        apply_prestate_trace(prestate_trace, ecx);
856
857        result
858    }
859}
