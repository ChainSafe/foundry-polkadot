210    /// Returns sources which include any tests to be executed.
211    /// If no filters are provided, sources are filtered by existence of test/invariant methods in
212    /// them, If filters are provided, sources are additionally filtered by them.
213    pub fn get_sources_to_compile(
214        &self,
215        config: &Config,
216        filter: &ProjectPathsAwareFilter,
217    ) -> Result<BTreeSet<PathBuf>> {
218        let mut project = config.create_project(true, true)?;
219        project.update_output_selection(|selection| {
220            *selection = OutputSelection::common_output_selection(["abi".to_string()]);
221        });
222
223        let output = project.compile()?;
224
225        if output.has_compiler_errors() {
226            sh_println!("{output}")?;
227            eyre::bail!("Compilation failed");
228        }
229
230        // ABIs of all sources
231        let abis = output
232            .into_artifacts()
233            .filter_map(|(id, artifact)| artifact.abi.map(|abi| (id, abi)))
234            .collect::<BTreeMap<_, _>>();
235
236        // Filter sources by their abis and contract names.
237        let mut test_sources = abis
238            .iter()
239            .filter(|(id, abi)| matches_contract(id, abi, filter))
240            .map(|(id, _)| id.source.clone())
241            .collect::<BTreeSet<_>>();
242
243        if test_sources.is_empty() {
244            if filter.is_empty() {
245                sh_println!(
246                    "No tests found in project! \
247                        Forge looks for functions that starts with `test`."
248                )?;
249            } else {
250                sh_println!("No tests match the provided pattern:")?;
251                sh_print!("{filter}")?;
252
253                // Try to suggest a test when there's no match
254                if let Some(test_pattern) = &filter.args().test_pattern {
255                    let test_name = test_pattern.as_str();
256                    let candidates = abis
257                        .into_iter()
258                        .filter(|(id, _)| {
259                            filter.matches_path(&id.source) && filter.matches_contract(&id.name)
260                        })
261                        .flat_map(|(_, abi)| abi.functions.into_keys())
262                        .collect::<Vec<_>>();
263                    if let Some(suggestion) = utils::did_you_mean(test_name, candidates).pop() {
264                        sh_println!("\nDid you mean `{suggestion}`?")?;
265                    }
266                }
267            }
268
269            eyre::bail!("No tests to run");
270        }
271
272        // Always recompile all sources to ensure that `getCode` cheatcode can use any artifact.
273        test_sources.extend(source_files_iter(
274            &project.paths.sources,
275            MultiCompilerLanguage::FILE_EXTENSIONS,
276        ));
277
278        Ok(test_sources)
279    }

281    /// Executes all the tests in the project.
282    ///
283    /// This will trigger the build process first. On success all test contracts that match the
284    /// configured filter will be executed
285    ///
286    /// Returns the test results for all matching tests.
287    pub async fn execute_tests(mut self) -> Result<TestOutcome> {
288        // Merge all configs.
289        let (mut config, mut evm_opts) = self.load_config_and_evm_opts()?;
290
291        let mut strategy = utils::get_executor_strategy(&config);
292
293        // Explicitly enable isolation for gas reports for more correct gas accounting.
294        if self.gas_report {
295            evm_opts.isolate = true;
296        } else {
297            // Do not collect gas report traces if gas report is not enabled.
298            config.fuzz.gas_report_samples = 0;
299            config.invariant.gas_report_samples = 0;
300        }
301
302        // Install missing dependencies.
303        if install::install_missing_dependencies(&mut config) && config.auto_detect_remappings {
304            // need to re-configure here to also catch additional remappings
305            config = self.load_config()?;
306        }
307
308        // Set up the project.
309        let project = config.project()?;
310
311        let filter = self.filter(&config)?;
312        trace!(target: "forge::test", ?filter, "using filter");
313
314        let sources_to_compile = self.get_sources_to_compile(&config, &filter)?;
315
316        // Handle compilation based on whether dual compilation is enabled
317        let (output, dual_compiled_contracts) = if config.resolc.resolc_compile {
318            // Dual compilation mode: compile both solc and resolc
319
320            // Compile with solc to a subdirectory
321            let mut solc_config = config.clone();
322            solc_config.out = solc_config.out.join(revive::SOLC_ARTIFACTS_SUBDIR);
323            solc_config.resolc = Default::default();
324            solc_config.build_info_path = Some(solc_config.out.join("build-info"));
325            let solc_project = solc_config.project()?;
326
327            let compiler = ProjectCompiler::new()
328                .dynamic_test_linking(config.dynamic_test_linking)
329                .quiet(shell::is_json() || self.junit)
330                .files(sources_to_compile.clone());
331
332            let solc_output = compiler.compile(&solc_project)?;
333
334            // Compile with resolc to the main output directory
335            let resolc_project = config.clone().project()?;
336
337            let resolc_compiler = ProjectCompiler::new()
338                .quiet(shell::is_json() || self.junit)
339                .files(sources_to_compile)
340                .size_limits(revive::CONTRACT_SIZE_LIMIT, revive::CONTRACT_SIZE_LIMIT);
341
342            let resolc_output = resolc_compiler.compile(&resolc_project)?;
343
344            // Create dual compiled contracts
345            let dual_compiled_contracts = DualCompiledContracts::new(
346                &solc_output,
347                &resolc_output,
348                &solc_project.paths,
349                &resolc_project.paths,
350            );
351
352            (solc_output, Some(dual_compiled_contracts))
353        } else {
354            // Single compilation mode: compile only with solc
355
356            let compiler: ProjectCompiler = ProjectCompiler::new()
357                .dynamic_test_linking(config.dynamic_test_linking)
358                .quiet(shell::is_json() || self.junit)
359                .files(sources_to_compile.clone());
360
361            let solc_output = compiler.compile(&project)?;
362
363            (solc_output, None)
364        };
365
366        // Create test options from general project settings and compiler output.
367        let project_root = &project.paths.root;
368
369        let should_debug = self.debug;
370        let should_draw = self.flamegraph || self.flamechart;
371
372        // Determine print verbosity and executor verbosity.
373        let verbosity = evm_opts.verbosity;
374        if (self.gas_report && evm_opts.verbosity < 3) || self.flamegraph || self.flamechart {
375            evm_opts.verbosity = 3;
376        }
377
378        let env = evm_opts.evm_env().await?;
379
380        // Enable internal tracing for more informative flamegraph.
381        if should_draw && !self.decode_internal {
382            self.decode_internal = true;
383        }
384
385        // Choose the internal function tracing mode, if --decode-internal is provided.
386        let decode_internal = if self.decode_internal {
387            // If more than one function matched, we enable simple tracing.
388            // If only one function matched, we enable full tracing. This is done in `run_tests`.
389            InternalTraceMode::Simple
390        } else {
391            InternalTraceMode::None
392        };
393
394        // Prepare the test builder.
395        let config = Arc::new(config);
396
397        // Set dual compiled contracts on the strategy
398        strategy.runner.revive_set_dual_compiled_contracts(
399            strategy.context.as_mut(),
400            dual_compiled_contracts.unwrap_or_default(),
401        );
402
403        let runner = MultiContractRunnerBuilder::new(config.clone())
404            .set_debug(should_debug)
405            .set_decode_internal(decode_internal)
406            .initial_balance(evm_opts.initial_balance)
407            .evm_spec(config.evm_spec_id())
408            .sender(evm_opts.sender)
409            .with_fork(evm_opts.get_fork(&config, env.clone()))
410            .enable_isolation(evm_opts.isolate)
411            .odyssey(evm_opts.odyssey)
412            .build::<MultiCompiler>(strategy, project_root, &output, env, evm_opts)?;
413
414        let libraries = runner.libraries.clone();
415        let mut outcome = self.run_tests(runner, config, verbosity, &filter, &output).await?;
416
417        if should_draw {
418            let (suite_name, test_name, mut test_result) =
419                outcome.remove_first().ok_or_eyre("no tests were executed")?;
420
421            let (_, arena) = test_result
422                .traces
423                .iter_mut()
424                .find(|(kind, _)| *kind == TraceKind::Execution)
425                .unwrap();
426
427            // Decode traces.
428            let decoder = outcome.last_run_decoder.as_ref().unwrap();
429            decode_trace_arena(arena, decoder).await;
430            let mut fst = folded_stack_trace::build(arena);
431
432            let label = if self.flamegraph { "flamegraph" } else { "flamechart" };
433            let contract = suite_name.split(':').next_back().unwrap();
434            let test_name = test_name.trim_end_matches("()");
435            let file_name = format!("cache/{label}_{contract}_{test_name}.svg");
436            let file = std::fs::File::create(&file_name).wrap_err("failed to create file")?;
437            let file = std::io::BufWriter::new(file);
438
439            let mut options = inferno::flamegraph::Options::default();
440            options.title = format!("{label} {contract}::{test_name}");
441            options.count_name = "gas".to_string();
442            if self.flamechart {
443                options.flame_chart = true;
444                fst.reverse();
445            }
446
447            // Generate SVG.
448            inferno::flamegraph::from_lines(&mut options, fst.iter().map(String::as_str), file)
449                .wrap_err("failed to write svg")?;
450            sh_println!("Saved to {file_name}")?;
451
452            // Open SVG in default program.
453            if let Err(e) = opener::open(&file_name) {
454                sh_err!("Failed to open {file_name}; please open it manually: {e}")?;
455            }
456        }
457
458        if should_debug {
459            // Get first non-empty suite result. We will have only one such entry.
460            let (_, _, test_result) =
461                outcome.remove_first().ok_or_eyre("no tests were executed")?;
462
463            let sources =
464                ContractSources::from_project_output(&output, project.root(), Some(&libraries))?;
465
466            // Run the debugger.
467            let mut builder = Debugger::builder()
468                .traces(
469                    test_result.traces.iter().filter(|(t, _)| t.is_execution()).cloned().collect(),
470                )
471                .sources(sources)
472                .breakpoints(test_result.breakpoints.clone());
473
474            if let Some(decoder) = &outcome.last_run_decoder {
475                builder = builder.decoder(decoder);
476            }
477
478            let mut debugger = builder.build();
479            if let Some(dump_path) = self.dump {
480                debugger.dump_to_file(&dump_path)?;
481            } else {
482                debugger.try_run_tui()?;
483            }
484        }
485
486        Ok(outcome)
487    }

489    /// Run all tests that matches the filter predicate from a test runner
490    pub async fn run_tests(
491        &self,
492        mut runner: MultiContractRunner,
493        config: Arc<Config>,
494        verbosity: u8,
495        filter: &ProjectPathsAwareFilter,
496        output: &ProjectCompileOutput,
497    ) -> eyre::Result<TestOutcome> {
498        if self.list {
499            return list(runner, filter);
500        }
501
502        trace!(target: "forge::test", "running all tests");
503
504        // If we need to render to a serialized format, we should not print anything else to stdout.
505        let silent = self.gas_report && shell::is_json() || self.summary && shell::is_json();
506
507        let num_filtered = runner.matching_test_functions(filter).count();
508        if num_filtered != 1 && (self.debug || self.flamegraph || self.flamechart) {
509            let action = if self.flamegraph {
510                "generate a flamegraph"
511            } else if self.flamechart {
512                "generate a flamechart"
513            } else {
514                "run the debugger"
515            };
516            let filter = if filter.is_empty() {
517                String::new()
518            } else {
519                format!("\n\nFilter used:\n{filter}")
520            };
521            eyre::bail!(
522                "{num_filtered} tests matched your criteria, but exactly 1 test must match in order to {action}.\n\n\
523                 Use --match-contract and --match-path to further limit the search.{filter}",
524            );
525        }
526
527        // If exactly one test matched, we enable full tracing.
528        if num_filtered == 1 && self.decode_internal {
529            runner.decode_internal = InternalTraceMode::Full;
530        }
531
532        // Run tests in a non-streaming fashion and collect results for serialization.
533        if !self.gas_report && !self.summary && shell::is_json() {
534            let mut results = runner.test_collect(filter)?;
535            results.values_mut().for_each(|suite_result| {
536                for test_result in suite_result.test_results.values_mut() {
537                    if verbosity >= 2 {
538                        // Decode logs at level 2 and above.
539                        test_result.decoded_logs = decode_console_logs(&test_result.logs);
540                    } else {
541                        // Empty logs for non verbose runs.
542                        test_result.logs = vec![];
543                    }
544                }
545            });
546            sh_println!("{}", serde_json::to_string(&results)?)?;
547            return Ok(TestOutcome::new(results, self.allow_failure));
548        }
549
550        if self.junit {
551            let results = runner.test_collect(filter)?;
552            sh_println!("{}", junit_xml_report(&results, verbosity).to_string()?)?;
553            return Ok(TestOutcome::new(results, self.allow_failure));
554        }
555
556        let remote_chain_id = runner.evm_opts.get_remote_chain_id().await;
557        let known_contracts = runner.known_contracts.clone();
558
559        let libraries = runner.libraries.clone();
560
561        // Run tests in a streaming fashion.
562        let (tx, rx) = channel::<(String, SuiteResult)>();
563        let timer = Instant::now();
564        let show_progress = config.show_progress;
565        let handle = tokio::task::spawn_blocking({
566            let filter = filter.clone();
567            move || runner.test(&filter, tx, show_progress)
568        });
569
570        // Set up trace identifiers.
571        let mut identifier = TraceIdentifiers::new().with_local(&known_contracts);
572
573        // Avoid using etherscan for gas report as we decode more traces and this will be
574        // expensive.
575        if !self.gas_report {
576            identifier = identifier.with_etherscan(&config, remote_chain_id)?;
577        }
578
579        // Build the trace decoder.
580        let mut builder = CallTraceDecoderBuilder::new()
581            .with_known_contracts(&known_contracts)
582            .with_verbosity(verbosity);
583        // Signatures are of no value for gas reports.
584        if !self.gas_report {
585            builder =
586                builder.with_signature_identifier(SignaturesIdentifier::from_config(&config)?);
587        }
588
589        if self.decode_internal {
590            let sources =
591                ContractSources::from_project_output(output, &config.root, Some(&libraries))?;
592            builder = builder.with_debug_identifier(DebugTraceIdentifier::new(sources));
593        }
594        let mut decoder = builder.build();
595
596        let mut gas_report = self.gas_report.then(|| {
597            GasReport::new(
598                config.gas_reports.clone(),
599                config.gas_reports_ignore.clone(),
600                config.gas_reports_include_tests,
601            )
602        });
603
604        let mut gas_snapshots = BTreeMap::<String, BTreeMap<String, String>>::new();
605
606        let mut outcome = TestOutcome::empty(self.allow_failure);
607
608        let mut any_test_failed = false;
609        for (contract_name, suite_result) in rx {
610            let tests = &suite_result.test_results;
611
612            // Clear the addresses and labels from previous test.
613            decoder.clear_addresses();
614
615            // We identify addresses if we're going to print *any* trace or gas report.
616            let identify_addresses = verbosity >= 3
617                || self.gas_report
618                || self.debug
619                || self.flamegraph
620                || self.flamechart;
621
622            // Print suite header.
623            if !silent {
624                sh_println!()?;
625                for warning in &suite_result.warnings {
626                    sh_warn!("{warning}")?;
627                }
628                if !tests.is_empty() {
629                    let len = tests.len();
630                    let tests = if len > 1 { "tests" } else { "test" };
631                    sh_println!("Ran {len} {tests} for {contract_name}")?;
632                }
633            }
634
635            // Process individual test results, printing logs and traces when necessary.
636            for (name, result) in tests {
637                let show_traces =
638                    !self.suppress_successful_traces || result.status == TestStatus::Failure;
639                if !silent {
640                    sh_println!("{}", result.short_result(name))?;
641
642                    // Display invariant metrics if invariant kind.
643                    if let TestKind::Invariant { metrics, .. } = &result.kind
644                        && !metrics.is_empty()
645                    {
646                        let _ = sh_println!("\n{}\n", format_invariant_metrics_table(metrics));
647                    }
648
649                    // We only display logs at level 2 and above
650                    if verbosity >= 2 && show_traces {
651                        // We only decode logs from Hardhat and DS-style console events
652                        let console_logs = decode_console_logs(&result.logs);
653                        if !console_logs.is_empty() {
654                            sh_println!("Logs:")?;
655                            for log in console_logs {
656                                sh_println!("  {log}")?;
657                            }
658                            sh_println!()?;
659                        }
660                    }
661                }
662
663                // We shouldn't break out of the outer loop directly here so that we finish
664                // processing the remaining tests and print the suite summary.
665                any_test_failed |= result.status == TestStatus::Failure;
666
667                // Clear the addresses and labels from previous runs.
668                decoder.clear_addresses();
669                decoder
670                    .labels
671                    .extend(result.labeled_addresses.iter().map(|(k, v)| (*k, v.clone())));
672
673                // Identify addresses and decode traces.
674                let mut decoded_traces = Vec::with_capacity(result.traces.len());
675                for (kind, arena) in &mut result.traces.clone() {
676                    if identify_addresses {
677                        decoder.identify(arena, &mut identifier);
678                    }
679
680                    // verbosity:
681                    // - 0..3: nothing
682                    // - 3: only display traces for failed tests
683                    // - 4: also display the setup trace for failed tests
684                    // - 5..: display all traces for all tests, including storage changes
685                    let should_include = match kind {
686                        TraceKind::Execution => {
687                            (verbosity == 3 && result.status.is_failure()) || verbosity >= 4
688                        }
689                        TraceKind::Setup => {
690                            (verbosity == 4 && result.status.is_failure()) || verbosity >= 5
691                        }
692                        TraceKind::Deployment => false,
693                    };
694
695                    if should_include {
696                        decode_trace_arena(arena, &decoder).await;
697                        decoded_traces.push(render_trace_arena_inner(arena, false, verbosity > 4));
698                    }
699                }
700
701                if !silent && show_traces && !decoded_traces.is_empty() {
702                    sh_println!("Traces:")?;
703                    for trace in &decoded_traces {
704                        sh_println!("{trace}")?;
705                    }
706                }
707
708                if let Some(gas_report) = &mut gas_report {
709                    gas_report.analyze(result.traces.iter().map(|(_, a)| &a.arena), &decoder).await;
710
711                    for trace in &result.gas_report_traces {
712                        decoder.clear_addresses();
713
714                        // Re-execute setup and deployment traces to collect identities created in
715                        // setUp and constructor.
716                        for (kind, arena) in &result.traces {
717                            if !matches!(kind, TraceKind::Execution) {
718                                decoder.identify(arena, &mut identifier);
719                            }
720                        }
721
722                        for arena in trace {
723                            decoder.identify(arena, &mut identifier);
724                            gas_report.analyze([arena], &decoder).await;
725                        }
726                    }
727                }
728
729                // Collect and merge gas snapshots.
730                for (group, new_snapshots) in &result.gas_snapshots {
731                    gas_snapshots.entry(group.clone()).or_default().extend(new_snapshots.clone());
732                }
733            }
734
735            // Write gas snapshots to disk if any were collected.
736            if !gas_snapshots.is_empty() {
737                // By default `gas_snapshot_check` is set to `false` in the config.
738                //
739                // The user can either:
740                // - Set `FORGE_SNAPSHOT_CHECK=true` in the environment.
741                // - Pass `--gas-snapshot-check=true` as a CLI argument.
742                // - Set `gas_snapshot_check = true` in the config.
743                //
744                // If the user passes `--gas-snapshot-check=<bool>` then it will override the config
745                // and the environment variable, disabling the check if `false` is passed.
746                //
747                // Exiting early with code 1 if differences are found.
748                if self.gas_snapshot_check.unwrap_or(config.gas_snapshot_check) {
749                    let differences_found = gas_snapshots.clone().into_iter().fold(
750                        false,
751                        |mut found, (group, snapshots)| {
752                            // If the snapshot file doesn't exist, we can't compare so we skip.
753                            if !&config.snapshots.join(format!("{group}.json")).exists() {
754                                return false;
755                            }
756
757                            let previous_snapshots: BTreeMap<String, String> =
758                                fs::read_json_file(&config.snapshots.join(format!("{group}.json")))
759                                    .expect("Failed to read snapshots from disk");
760
761                            let diff: BTreeMap<_, _> = snapshots
762                                .iter()
763                                .filter_map(|(k, v)| {
764                                    previous_snapshots.get(k).and_then(|previous_snapshot| {
765                                        if previous_snapshot != v {
766                                            Some((
767                                                k.clone(),
768                                                (previous_snapshot.clone(), v.clone()),
769                                            ))
770                                        } else {
771                                            None
772                                        }
773                                    })
774                                })
775                                .collect();
776
777                            if !diff.is_empty() {
778                                let _ = sh_eprintln!(
779                                    "{}",
780                                    format!("\n[{group}] Failed to match snapshots:").red().bold()
781                                );
782
783                                for (key, (previous_snapshot, snapshot)) in &diff {
784                                    let _ = sh_eprintln!(
785                                        "{}",
786                                        format!("- [{key}] {previous_snapshot}  {snapshot}").red()
787                                    );
788                                }
789
790                                found = true;
791                            }
792
793                            found
794                        },
795                    );
796
797                    if differences_found {
798                        sh_eprintln!()?;
799                        eyre::bail!("Snapshots differ from previous run");
800                    }
801                }
802
803                // By default `gas_snapshot_emit` is set to `true` in the config.
804                //
805                // The user can either:
806                // - Set `FORGE_SNAPSHOT_EMIT=false` in the environment.
807                // - Pass `--gas-snapshot-emit=false` as a CLI argument.
808                // - Set `gas_snapshot_emit = false` in the config.
809                //
810                // If the user passes `--gas-snapshot-emit=<bool>` then it will override the config
811                // and the environment variable, enabling the check if `true` is passed.
812                if self.gas_snapshot_emit.unwrap_or(config.gas_snapshot_emit) {
813                    // Create `snapshots` directory if it doesn't exist.
814                    fs::create_dir_all(&config.snapshots)?;
815
816                    // Write gas snapshots to disk per group.
817                    gas_snapshots.clone().into_iter().for_each(|(group, snapshots)| {
818                        fs::write_pretty_json_file(
819                            &config.snapshots.join(format!("{group}.json")),
820                            &snapshots,
821                        )
822                        .expect("Failed to write gas snapshots to disk");
823                    });
824                }
825            }
826
827            // Print suite summary.
828            if !silent {
829                sh_println!("{}", suite_result.summary())?;
830            }
831
832            // Add the suite result to the outcome.
833            outcome.results.insert(contract_name, suite_result);
834
835            // Stop processing the remaining suites if any test failed and `fail_fast` is set.
836            if self.fail_fast && any_test_failed {
837                break;
838            }
839        }
840        outcome.last_run_decoder = Some(decoder);
841        let duration = timer.elapsed();
842
843        trace!(target: "forge::test", len=outcome.results.len(), %any_test_failed, "done with results");
844
845        if let Some(gas_report) = gas_report {
846            let finalized = gas_report.finalize();
847            sh_println!("{}", &finalized)?;
848            outcome.gas_report = Some(finalized);
849        }
850
851        if !self.summary && !shell::is_json() {
852            sh_println!("{}", outcome.summary(duration))?;
853        }
854
855        if self.summary && !outcome.results.is_empty() {
856            let summary_report = TestSummaryReport::new(self.detailed, outcome.clone());
857            sh_println!("{}", &summary_report)?;
858        }
859
860        // Reattach the task.
861        if let Err(e) = handle.await {
862            match e.try_into_panic() {
863                Ok(payload) => std::panic::resume_unwind(payload),
864                Err(e) => return Err(e.into()),
865            }
866        }
867
868        // Persist test run failures to enable replaying.
869        persist_run_failures(&config, &outcome);
870
871        Ok(outcome)
872    }

874    /// Returns the flattened [`FilterArgs`] arguments merged with [`Config`].
875    /// Loads and applies filter from file if only last test run failures performed.
876    pub fn filter(&self, config: &Config) -> Result<ProjectPathsAwareFilter> {
877        let mut filter = self.filter.clone();
878        if self.rerun {
879            filter.test_pattern = last_run_failures(config);
880        }
881        if filter.path_pattern.is_some() {
882            if self.path.is_some() {
883                bail!("Can not supply both --match-path and |path|");
884            }
885        } else {
886            filter.path_pattern = self.path.clone();
887        }
888        Ok(filter.merge_with_config(config))
889    }

892    pub fn is_watch(&self) -> bool {
893        self.watch.watch.is_some()
894    }

896    /// Returns the [`watchexec::Config`] necessary to bootstrap a new watch loop.
897    pub(crate) fn watchexec_config(&self) -> Result<watchexec::Config> {
898        self.watch.watchexec_config(|| {
899            let config = self.load_config()?;
900            let foundry_toml: PathBuf = config.root.join(Config::FILE_NAME);
901            Ok([config.src, config.test, config.script, foundry_toml])
902        })
903    }

907    fn metadata(&self) -> Metadata {
908        Metadata::named("Core Build Args Provider")
909    }
910
911    fn data(&self) -> Result<Map<Profile, Dict>, figment::Error> {
912        let mut dict = Dict::default();
913
914        let mut fuzz_dict = Dict::default();
915        if let Some(fuzz_seed) = self.fuzz_seed {
916            fuzz_dict.insert("seed".to_string(), fuzz_seed.to_string().into());
917        }
918        if let Some(fuzz_runs) = self.fuzz_runs {
919            fuzz_dict.insert("runs".to_string(), fuzz_runs.into());
920        }
921        if let Some(fuzz_timeout) = self.fuzz_timeout {
922            fuzz_dict.insert("timeout".to_string(), fuzz_timeout.into());
923        }
924        if let Some(fuzz_input_file) = self.fuzz_input_file.clone() {
925            fuzz_dict.insert("failure_persist_file".to_string(), fuzz_input_file.into());
926        }
927        dict.insert("fuzz".to_string(), fuzz_dict.into());
928
929        if let Some(etherscan_api_key) =
930            self.etherscan_api_key.as_ref().filter(|s| !s.trim().is_empty())
931        {
932            dict.insert("etherscan_api_key".to_string(), etherscan_api_key.to_string().into());
933        }
934
935        if let Some(api_version) = &self.etherscan_api_version {
936            dict.insert("etherscan_api_version".to_string(), api_version.to_string().into());
937        }
938
939        if self.show_progress {
940            dict.insert("show_progress".to_string(), true.into());
941        }
942
943        Ok(Map::from([(Config::selected_profile(), dict)]))
944    }

947/// Lists all matching tests
948fn list(runner: MultiContractRunner, filter: &ProjectPathsAwareFilter) -> Result<TestOutcome> {
949    let results = runner.list(filter);
950
951    if shell::is_json() {
952        sh_println!("{}", serde_json::to_string(&results)?)?;
953    } else {
954        for (file, contracts) in &results {
955            sh_println!("{file}")?;
956            for (contract, tests) in contracts {
957                sh_println!("  {contract}")?;
958                sh_println!("    {}\n", tests.join("\n    "))?;
959            }
960        }
961    }
962    Ok(TestOutcome::empty(false))
963}

965/// Load persisted filter (with last test run failures) from file.
966fn last_run_failures(config: &Config) -> Option<regex::Regex> {
967    match fs::read_to_string(&config.test_failures_file) {
968        Ok(filter) => Some(Regex::new(&filter).unwrap()),
969        Err(_) => None,
970    }
971}
972
973/// Persist filter with last test run failures (only if there's any failure).
974fn persist_run_failures(config: &Config, outcome: &TestOutcome) {
975    if outcome.failed() > 0 && fs::create_file(&config.test_failures_file).is_ok() {
976        let mut filter = String::new();
977        let mut failures = outcome.failures().peekable();
978        while let Some((test_name, _)) = failures.next() {
979            if test_name.is_any_test()
980                && let Some(test_match) = test_name.split("(").next()
981            {
982                filter.push_str(test_match);
983                if failures.peek().is_some() {
984                    filter.push('|');
985                }
986            }
987        }
988        let _ = fs::write(&config.test_failures_file, filter);
989    }
990}

992/// Generate test report in JUnit XML report format.
993fn junit_xml_report(results: &BTreeMap<String, SuiteResult>, verbosity: u8) -> Report {
994    let mut total_duration = Duration::default();
995    let mut junit_report = Report::new("Test run");
996    junit_report.set_timestamp(Utc::now());
997    for (suite_name, suite_result) in results {
998        let mut test_suite = TestSuite::new(suite_name);
999        total_duration += suite_result.duration;
1000        test_suite.set_time(suite_result.duration);
1001        test_suite.set_system_out(suite_result.summary());
1002        for (test_name, test_result) in &suite_result.test_results {
1003            let mut test_status = match test_result.status {
1004                TestStatus::Success => TestCaseStatus::success(),
1005                TestStatus::Failure => TestCaseStatus::non_success(NonSuccessKind::Failure),
1006                TestStatus::Skipped => TestCaseStatus::skipped(),
1007            };
1008            if let Some(reason) = &test_result.reason {
1009                test_status.set_message(reason);
1010            }
1011
1012            let mut test_case = TestCase::new(test_name, test_status);
1013            test_case.set_time(test_result.duration);
1014
1015            let mut sys_out = String::new();
1016            let result_report = test_result.kind.report();
1017            write!(sys_out, "{test_result} {test_name} {result_report}").unwrap();
1018            if verbosity >= 2 && !test_result.logs.is_empty() {
1019                write!(sys_out, "\\nLogs:\\n").unwrap();
1020                let console_logs = decode_console_logs(&test_result.logs);
1021                for log in console_logs {
1022                    write!(sys_out, "  {log}\\n").unwrap();
1023                }
1024            }
1025
1026            test_case.set_system_out(sys_out);
1027            test_suite.add_test_case(test_case);
1028        }
1029        junit_report.add_test_suite(test_suite);
1030    }
1031    junit_report.set_time(total_duration);
1032    junit_report
1033}
