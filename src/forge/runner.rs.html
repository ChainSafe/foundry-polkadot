107    /// Deploys the test contract inside the runner from the sending account, and optionally runs
108    /// the `setUp` function on the test contract.
109    pub fn setup(&mut self, call_setup: bool) -> TestSetup {
110        self._setup(call_setup).unwrap_or_else(|err| {
111            if err.to_string().contains("skipped") {
112                TestSetup::skipped(err.to_string())
113            } else {
114                TestSetup::failed(err.to_string())
115            }
116        })
117    }
118
119    fn _setup(&mut self, call_setup: bool) -> Result<TestSetup> {
120        trace!(call_setup, "setting up");
121
122        self.apply_contract_inline_config()?;
123
124        // We max out their balance so that they can deploy and make calls.
125        self.executor.set_balance(self.sender, U256::MAX)?;
126        self.executor.set_balance(CALLER, U256::MAX)?;
127
128        // We set the nonce of the deployer accounts to 1 to get the same addresses as DappTools.
129        self.executor.set_nonce(self.sender, 1)?;
130
131        // Deploy libraries.
132        self.executor.set_balance(LIBRARY_DEPLOYER, U256::MAX)?;
133
134        let mut result = TestSetup::default();
135        for code in &self.mcr.libs_to_deploy {
136            let deploy_result = self.executor.deploy(
137                LIBRARY_DEPLOYER,
138                code.clone(),
139                U256::ZERO,
140                Some(&self.mcr.revert_decoder),
141            );
142
143            // Record deployed library address.
144            if let Ok(deployed) = &deploy_result {
145                result.deployed_libs.push(deployed.address);
146            }
147
148            let (raw, reason) = RawCallResult::from_evm_result(deploy_result.map(Into::into))?;
149            result.extend(raw, TraceKind::Deployment);
150            if reason.is_some() {
151                result.reason = reason;
152                return Ok(result);
153            }
154        }
155
156        let address = self.sender.create(self.executor.get_nonce(self.sender)?);
157        result.address = address;
158        // NOTE(revive): the test contract is set here instead of where upstream does it as
159        // the test contract address needs to be retrieved in order to skip
160        // revive mode for the creation of the test address (and for calls to it later).
161        self.executor.backend_mut().set_test_contract(address);
162
163        // Set the contracts initial balance before deployment, so it is available during
164        // construction
165        self.executor.set_balance(address, self.initial_balance())?;
166
167        // Deploy the test contract
168        let deploy_result = self.executor.deploy(
169            self.sender,
170            self.contract.bytecode.clone(),
171            U256::ZERO,
172            Some(&self.mcr.revert_decoder),
173        );
174
175        result.deployment_failure = deploy_result.is_err();
176
177        if let Ok(dr) = &deploy_result {
178            debug_assert_eq!(dr.address, address);
179        }
180        let (raw, reason) = RawCallResult::from_evm_result(deploy_result.map(Into::into))?;
181        result.extend(raw, TraceKind::Deployment);
182        if reason.is_some() {
183            result.reason = reason;
184            return Ok(result);
185        }
186
187        // Reset `self.sender`s, `CALLER`s and `LIBRARY_DEPLOYER`'s balance to the initial balance.
188        self.executor.set_balance(self.sender, self.initial_balance())?;
189        self.executor.set_balance(CALLER, self.initial_balance())?;
190        self.executor.set_balance(LIBRARY_DEPLOYER, self.initial_balance())?;
191
192        self.executor.deploy_create2_deployer()?;
193
194        if let Some(cheatcodes) = &mut self.executor.inspector_mut().cheatcodes {
195            debug!("test contract deployed");
196            cheatcodes.strategy.runner.base_contract_deployed(cheatcodes.strategy.context.as_mut());
197        }
198
199        // Optionally call the `setUp` function
200        if call_setup {
201            trace!("calling setUp");
202            let res = self.executor.setup(None, address, Some(&self.mcr.revert_decoder));
203            let (raw, reason) = RawCallResult::from_evm_result(res)?;
204            result.extend(raw, TraceKind::Setup);
205            result.reason = reason;
206        }
207
208        result.fuzz_fixtures = self.fuzz_fixtures(address);
209
210        Ok(result)
211    }

217    /// Configures this runner with the inline configuration for the contract.
218    fn apply_contract_inline_config(&mut self) -> Result<()> {
219        if self.inline_config.contains_contract(self.name) {
220            let new_config = Arc::new(self.inline_config(None)?);
221            self.tcfg.to_mut().reconfigure_with(new_config);
222            let prev_tracer = self.executor.inspector_mut().tracer.take();
223            self.tcfg.configure_executor(&mut self.executor);
224            // Don't set tracer here.
225            self.executor.inspector_mut().tracer = prev_tracer;
226        }
227        Ok(())
228    }

238    /// Collect fixtures from test contract.
239    ///
240    /// Fixtures can be defined:
241    /// - as storage arrays in test contract, prefixed with `fixture`
242    /// - as functions prefixed with `fixture` and followed by parameter name to be fuzzed
243    ///
244    /// Storage array fixtures:
245    /// `uint256[] public fixture_amount = [1, 2, 3];`
246    /// define an array of uint256 values to be used for fuzzing `amount` named parameter in scope
247    /// of the current test.
248    ///
249    /// Function fixtures:
250    /// `function fixture_owner() public returns (address[] memory){}`
251    /// returns an array of addresses to be used for fuzzing `owner` named parameter in scope of the
252    /// current test.
253    fn fuzz_fixtures(&mut self, address: Address) -> FuzzFixtures {
254        let mut fixtures = HashMap::default();
255        let fixture_functions = self.contract.abi.functions().filter(|func| func.is_fixture());
256        for func in fixture_functions {
257            if func.inputs.is_empty() {
258                // Read fixtures declared as functions.
259                if let Ok(CallResult { raw: _, decoded_result }) =
260                    self.executor.call(CALLER, address, func, &[], U256::ZERO, None)
261                {
262                    fixtures.insert(fixture_name(func.name.clone()), decoded_result);
263                }
264            } else {
265                // For reading fixtures from storage arrays we collect values by calling the
266                // function with incremented indexes until there's an error.
267                let mut vals = Vec::new();
268                let mut index = 0;
269                loop {
270                    if let Ok(CallResult { raw: _, decoded_result }) = self.executor.call(
271                        CALLER,
272                        address,
273                        func,
274                        &[DynSolValue::Uint(U256::from(index), 256)],
275                        U256::ZERO,
276                        None,
277                    ) {
278                        vals.push(decoded_result);
279                    } else {
280                        // No result returned for this index, we reached the end of storage
281                        // array or the function is not a valid fixture.
282                        break;
283                    }
284                    index += 1;
285                }
286                fixtures.insert(fixture_name(func.name.clone()), DynSolValue::Array(vals));
287            };
288        }
289        FuzzFixtures::new(fixtures)
290    }

292    /// Runs all tests for a contract whose names match the provided regular expression
293    pub fn run_tests(mut self, filter: &dyn TestFilter) -> SuiteResult {
294        let start = Instant::now();
295        let mut warnings = Vec::new();
296
297        // Check if `setUp` function with valid signature declared.
298        let setup_fns: Vec<_> =
299            self.contract.abi.functions().filter(|func| func.name.is_setup()).collect();
300        let call_setup = setup_fns.len() == 1 && setup_fns[0].name == "setUp";
301        // There is a single miss-cased `setUp` function, so we add a warning
302        for &setup_fn in &setup_fns {
303            if setup_fn.name != "setUp" {
304                warnings.push(format!(
305                    "Found invalid setup function \"{}\" did you mean \"setUp()\"?",
306                    setup_fn.signature()
307                ));
308            }
309        }
310
311        // There are multiple setUp function, so we return a single test result for `setUp`
312        if setup_fns.len() > 1 {
313            return SuiteResult::new(
314                start.elapsed(),
315                [("setUp()".to_string(), TestResult::fail("multiple setUp functions".to_string()))]
316                    .into(),
317                warnings,
318            );
319        }
320
321        // Check if `afterInvariant` function with valid signature declared.
322        let after_invariant_fns: Vec<_> =
323            self.contract.abi.functions().filter(|func| func.name.is_after_invariant()).collect();
324        if after_invariant_fns.len() > 1 {
325            // Return a single test result failure if multiple functions declared.
326            return SuiteResult::new(
327                start.elapsed(),
328                [(
329                    "afterInvariant()".to_string(),
330                    TestResult::fail("multiple afterInvariant functions".to_string()),
331                )]
332                .into(),
333                warnings,
334            );
335        }
336        let call_after_invariant = after_invariant_fns.first().is_some_and(|after_invariant_fn| {
337            let match_sig = after_invariant_fn.name == "afterInvariant";
338            if !match_sig {
339                warnings.push(format!(
340                    "Found invalid afterInvariant function \"{}\" did you mean \"afterInvariant()\"?",
341                    after_invariant_fn.signature()
342                ));
343            }
344            match_sig
345        });
346
347        // Invariant testing requires tracing to figure out what contracts were created.
348        // We also want to disable `debug` for setup since we won't be using those traces.
349        let has_invariants = self.contract.abi.functions().any(|func| func.is_invariant_test());
350
351        let prev_tracer = self.executor.inspector_mut().tracer.take();
352        if prev_tracer.is_some() || has_invariants {
353            self.executor.set_tracing(TraceMode::Call);
354        }
355
356        let setup_time = Instant::now();
357        let setup = self.setup(call_setup);
358        debug!("finished setting up in {:?}", setup_time.elapsed());
359
360        self.executor.inspector_mut().tracer = prev_tracer;
361
362        if setup.reason.is_some() {
363            // The setup failed, so we return a single test result for `setUp`
364            let fail_msg = if !setup.deployment_failure {
365                "setUp()".to_string()
366            } else {
367                "constructor()".to_string()
368            };
369            return SuiteResult::new(
370                start.elapsed(),
371                [(fail_msg, TestResult::setup_result(setup))].into(),
372                warnings,
373            );
374        }
375
376        // Filter out functions sequentially since it's very fast and there is no need to do it
377        // in parallel.
378        let find_timer = Instant::now();
379        let functions = self
380            .contract
381            .abi
382            .functions()
383            .filter(|func| is_matching_test(func, filter))
384            .collect::<Vec<_>>();
385        debug!(
386            "Found {} test functions out of {} in {:?}",
387            functions.len(),
388            self.contract.abi.functions().count(),
389            find_timer.elapsed(),
390        );
391
392        let identified_contracts = has_invariants.then(|| {
393            load_contracts(setup.traces.iter().map(|(_, t)| &t.arena), &self.mcr.known_contracts)
394        });
395
396        let test_fail_instances = functions
397            .iter()
398            .filter_map(|func| {
399                TestFunctionKind::classify(&func.name, !func.inputs.is_empty())
400                    .is_any_test_fail()
401                    .then_some(func.name.clone())
402            })
403            .collect::<Vec<_>>();
404
405        if !test_fail_instances.is_empty() {
406            let instances = format!(
407                "Found {} instances: {}",
408                test_fail_instances.len(),
409                test_fail_instances.join(", ")
410            );
411            let fail =  TestResult::fail("`testFail*` has been removed. Consider changing to test_Revert[If|When]_Condition and expecting a revert".to_string());
412            return SuiteResult::new(start.elapsed(), [(instances, fail)].into(), warnings);
413        }
414        let f = |backend: Option<revive_strategy::Backend>, func: &Function| {
415            let f = || {
416                let start = Instant::now();
417
418                let _guard = self.tokio_handle.enter();
419
420                let _guard;
421                let current_span = tracing::Span::current();
422                if current_span.is_none() || current_span.id() != self.span.id() {
423                    _guard = self.span.enter();
424                }
425
426                let sig = func.signature();
427                let kind = func.test_function_kind();
428
429                let _guard = debug_span!(
430                    "test",
431                    %kind,
432                    name = %if enabled!(tracing::Level::TRACE) { &sig } else { &func.name },
433                )
434                .entered();
435
436                let mut res = FunctionRunner::new(&self, &setup).run(
437                    func,
438                    kind,
439                    call_after_invariant,
440                    identified_contracts.as_ref(),
441                );
442                res.duration = start.elapsed();
443
444                (sig, res)
445            };
446            if let Some(backend) = backend {
447                revive_strategy::with_externalities(backend, f)
448            } else {
449                f()
450            }
451        };
452        let backend = if self.config.resolc.resolc_compile {
453            Some(revive_strategy::Backend::get())
454        } else {
455            None
456        };
457
458        let test_results = functions
459            .into_par_iter()
460            .map(|item| f(backend.clone(), item))
461            .collect::<BTreeMap<_, _>>();
462
463        let duration = start.elapsed();
464        SuiteResult::new(duration, test_results, warnings)
465    }
466}

468/// Executes a single test function, returning a [`TestResult`].
469struct FunctionRunner<'a> {
470    /// The function-level configuration.
471    tcfg: Cow<'a, TestRunnerConfig>,
472    /// The EVM executor.
473    executor: Cow<'a, Executor>,
474    /// The parent runner.
475    cr: &'a ContractRunner<'a>,
476    /// The address of the test contract.
477    address: Address,
478    /// The test setup result.
479    setup: &'a TestSetup,
480    /// The test result. Returned after running the test.
481    result: TestResult,
482}

493impl<'a> FunctionRunner<'a> {
494    fn new(cr: &'a ContractRunner<'a>, setup: &'a TestSetup) -> Self {
495        Self {
496            tcfg: match &cr.tcfg {
497                Cow::Borrowed(tcfg) => Cow::Borrowed(tcfg),
498                Cow::Owned(tcfg) => Cow::Owned(tcfg.clone()),
499            },
500            executor: Cow::Borrowed(&cr.executor),
501            cr,
502            address: setup.address,
503            setup,
504            result: TestResult::new(setup),
505        }
506    }
507
508    fn revert_decoder(&self) -> &'a RevertDecoder {
509        &self.cr.mcr.revert_decoder
510    }
511
512    /// Configures this runner with the inline configuration for the contract.
513    fn apply_function_inline_config(&mut self, func: &Function) -> Result<()> {
514        if self.inline_config.contains_function(self.cr.name, &func.name) {
515            let new_config = Arc::new(self.cr.inline_config(Some(func))?);
516            self.tcfg.to_mut().reconfigure_with(new_config);
517            self.tcfg.configure_executor(self.executor.to_mut());
518        }
519        Ok(())
520    }
521
522    fn run(
523        mut self,
524        func: &Function,
525        kind: TestFunctionKind,
526        call_after_invariant: bool,
527        identified_contracts: Option<&ContractsByAddress>,
528    ) -> TestResult {
529        if let Err(e) = self.apply_function_inline_config(func) {
530            self.result.single_fail(Some(e.to_string()));
531            return self.result;
532        }
533
534        match kind {
535            TestFunctionKind::UnitTest { .. } => self.run_unit_test(func),
536            TestFunctionKind::FuzzTest { .. } => self.run_fuzz_test(func),
537            TestFunctionKind::TableTest => self.run_table_test(func),
538            TestFunctionKind::InvariantTest => {
539                let test_bytecode = &self.cr.contract.bytecode;
540                self.run_invariant_test(
541                    func,
542                    call_after_invariant,
543                    identified_contracts.unwrap(),
544                    test_bytecode,
545                )
546            }
547            _ => unreachable!(),
548        }
549    }

551    /// Runs a single unit test.
552    ///
553    /// Applies before test txes (if any), runs current test and returns the `TestResult`.
554    ///
555    /// Before test txes are applied in order and state modifications committed to the EVM database
556    /// (therefore the unit test call will be made on modified state).
557    /// State modifications of before test txes and unit test function call are discarded after
558    /// test ends, similar to `eth_call`.
559    fn run_unit_test(mut self, func: &Function) -> TestResult {
560        // Prepare unit test execution.
561        if self.prepare_test(func).is_err() {
562            return self.result;
563        }
564        let mut binding = self.executor.clone();
565        let executor = binding.to_mut();
566        // Run current unit test.
567        let (mut raw_call_result, reason) = match executor.call(
568            self.sender,
569            self.address,
570            func,
571            &[],
572            U256::ZERO,
573            Some(self.revert_decoder()),
574        ) {
575            Ok(res) => (res.raw, None),
576            Err(EvmError::Execution(err)) => (err.raw, Some(err.reason)),
577            Err(EvmError::Skip(reason)) => {
578                self.result.single_skip(reason);
579                return self.result;
580            }
581            Err(err) => {
582                self.result.single_fail(Some(err.to_string()));
583                return self.result;
584            }
585        };
586
587        let success =
588            self.executor.is_raw_call_mut_success(self.address, &mut raw_call_result, false);
589        self.result.single_result(success, reason, raw_call_result);
590        self.result
591    }
592
593    /// Runs a table test.
594    /// The parameters dataset (table) is created from defined parameter fixtures, therefore each
595    /// test table parameter should have the same number of fixtures defined.
596    /// E.g. for table test
597    /// - `table_test(uint256 amount, bool swap)` fixtures are defined as
598    /// - `uint256[] public fixtureAmount = [2, 5]`
599    /// - `bool[] public fixtureSwap = [true, false]` The `table_test` is then called with the pair
600    ///   of args `(2, true)` and `(5, false)`.
601    fn run_table_test(mut self, func: &Function) -> TestResult {
602        // Prepare unit test execution.
603        if self.prepare_test(func).is_err() {
604            return self.result;
605        }
606
607        // Extract and validate fixtures for the first table test parameter.
608        let Some(first_param) = func.inputs.first() else {
609            self.result.single_fail(Some("Table test should have at least one parameter".into()));
610            return self.result;
611        };
612
613        let Some(first_param_fixtures) =
614            &self.setup.fuzz_fixtures.param_fixtures(first_param.name())
615        else {
616            self.result.single_fail(Some("Table test should have fixtures defined".into()));
617            return self.result;
618        };
619
620        if first_param_fixtures.is_empty() {
621            self.result.single_fail(Some("Table test should have at least one fixture".into()));
622            return self.result;
623        }
624
625        let fixtures_len = first_param_fixtures.len();
626        let mut table_fixtures = vec![&first_param_fixtures[..]];
627
628        // Collect fixtures for remaining parameters.
629        for param in &func.inputs[1..] {
630            let param_name = param.name();
631            let Some(fixtures) = &self.setup.fuzz_fixtures.param_fixtures(param.name()) else {
632                self.result.single_fail(Some(format!("No fixture defined for param {param_name}")));
633                return self.result;
634            };
635
636            if fixtures.len() != fixtures_len {
637                self.result.single_fail(Some(format!(
638                    "{} fixtures defined for {param_name} (expected {})",
639                    fixtures.len(),
640                    fixtures_len
641                )));
642                return self.result;
643            }
644
645            table_fixtures.push(&fixtures[..]);
646        }
647
648        let progress = start_fuzz_progress(
649            self.cr.progress,
650            self.cr.name,
651            &func.name,
652            None,
653            fixtures_len as u32,
654        );
655
656        for i in 0..fixtures_len {
657            // Increment progress bar.
658            if let Some(progress) = progress.as_ref() {
659                progress.inc(1);
660            }
661
662            let args = table_fixtures.iter().map(|row| row[i].clone()).collect_vec();
663            let (mut raw_call_result, reason) = match self.executor.call(
664                self.sender,
665                self.address,
666                func,
667                &args,
668                U256::ZERO,
669                Some(self.revert_decoder()),
670            ) {
671                Ok(res) => (res.raw, None),
672                Err(EvmError::Execution(err)) => (err.raw, Some(err.reason)),
673                Err(EvmError::Skip(reason)) => {
674                    self.result.single_skip(reason);
675                    return self.result;
676                }
677                Err(err) => {
678                    self.result.single_fail(Some(err.to_string()));
679                    return self.result;
680                }
681            };
682
683            let is_success =
684                self.executor.is_raw_call_mut_success(self.address, &mut raw_call_result, false);
685            // Record counterexample if test fails.
686            if !is_success {
687                self.result.counterexample =
688                    Some(CounterExample::Single(BaseCounterExample::from_fuzz_call(
689                        Bytes::from(func.abi_encode_input(&args).unwrap()),
690                        args,
691                        raw_call_result.traces.clone(),
692                    )));
693                self.result.single_result(false, reason, raw_call_result);
694                return self.result;
695            }
696
697            // If it's the last iteration and all other runs succeeded, then use last call result
698            // for logs and traces.
699            if i == fixtures_len - 1 {
700                self.result.single_result(true, None, raw_call_result);
701                return self.result;
702            }
703        }
704
705        self.result
706    }

708    fn run_invariant_test(
709        mut self,
710        func: &Function,
711        call_after_invariant: bool,
712        identified_contracts: &ContractsByAddress,
713        test_bytecode: &Bytes,
714    ) -> TestResult {
715        let mut binding = self.executor.clone();
716        let executor = binding.to_mut();
717        // First, run the test normally to see if it needs to be skipped.
718        if let Err(EvmError::Skip(reason)) = executor.call(
719            self.sender,
720            self.address,
721            func,
722            &[],
723            U256::ZERO,
724            Some(self.revert_decoder()),
725        ) {
726            self.result.invariant_skip(reason);
727            return self.result;
728        };
729
730        let runner = self.invariant_runner();
731        let invariant_config = &self.config.invariant;
732
733        let mut executor = self.clone_executor();
734        // Enable edge coverage if running with coverage guided fuzzing or with edge coverage
735        // metrics (useful for benchmarking the fuzzer).
736        executor.inspector_mut().collect_edge_coverage(
737            invariant_config.corpus_dir.is_some() || invariant_config.show_edge_coverage,
738        );
739
740        let mut evm = InvariantExecutor::new(
741            executor,
742            runner,
743            invariant_config.clone(),
744            identified_contracts,
745            &self.cr.mcr.known_contracts,
746        );
747        let invariant_contract = InvariantContract {
748            address: self.address,
749            invariant_function: func,
750            call_after_invariant,
751            abi: &self.cr.contract.abi,
752        };
753
754        let (failure_dir, failure_file) = invariant_failure_paths(
755            invariant_config,
756            self.cr.name,
757            &invariant_contract.invariant_function.name,
758        );
759        let show_solidity = invariant_config.show_solidity;
760
761        // Try to replay recorded failure if any.
762        if let Some(mut call_sequence) =
763            persisted_call_sequence(failure_file.as_path(), test_bytecode)
764        {
765            // Create calls from failed sequence and check if invariant still broken.
766            let txes = call_sequence
767                .iter_mut()
768                .map(|seq| {
769                    seq.show_solidity = show_solidity;
770                    BasicTxDetails {
771                        sender: seq.sender.unwrap_or_default(),
772                        call_details: CallDetails {
773                            target: seq.addr.unwrap_or_default(),
774                            calldata: seq.calldata.clone(),
775                        },
776                    }
777                })
778                .collect::<Vec<BasicTxDetails>>();
779            if let Ok((success, replayed_entirely)) = check_sequence(
780                self.clone_executor(),
781                &txes,
782                (0..min(txes.len(), invariant_config.depth as usize)).collect(),
783                invariant_contract.address,
784                invariant_contract.invariant_function.selector().to_vec().into(),
785                invariant_config.fail_on_revert,
786                invariant_contract.call_after_invariant,
787            ) && !success
788            {
789                let _ = sh_warn!(
790                    "\
791                            Replayed invariant failure from {:?} file. \
792                            Run `forge clean` or remove file to ignore failure and to continue invariant test campaign.",
793                    failure_file.as_path()
794                );
795                // If sequence still fails then replay error to collect traces and
796                // exit without executing new runs.
797                let _ = replay_run(
798                    &invariant_contract,
799                    self.clone_executor(),
800                    &self.cr.mcr.known_contracts,
801                    identified_contracts.clone(),
802                    &mut self.result.logs,
803                    &mut self.result.traces,
804                    &mut self.result.line_coverage,
805                    &mut self.result.deprecated_cheatcodes,
806                    &txes,
807                    show_solidity,
808                );
809                self.result.invariant_replay_fail(
810                    replayed_entirely,
811                    &invariant_contract.invariant_function.name,
812                    call_sequence,
813                );
814                return self.result;
815            }
816        }
817
818        let progress = start_fuzz_progress(
819            self.cr.progress,
820            self.cr.name,
821            &func.name,
822            invariant_config.timeout,
823            invariant_config.runs,
824        );
825        let invariant_result = match evm.invariant_fuzz(
826            invariant_contract.clone(),
827            &self.setup.fuzz_fixtures,
828            &self.setup.deployed_libs,
829            progress.as_ref(),
830        ) {
831            Ok(x) => x,
832            Err(e) => {
833                self.result.invariant_setup_fail(e);
834                return self.result;
835            }
836        };
837        // Merge coverage collected during invariant run with test setup coverage.
838        self.result.merge_coverages(invariant_result.line_coverage);
839
840        let mut counterexample = None;
841        let success = invariant_result.error.is_none();
842        let reason = invariant_result.error.as_ref().and_then(|err| err.revert_reason());
843
844        match invariant_result.error {
845            // If invariants were broken, replay the error to collect logs and traces
846            Some(error) => match error {
847                InvariantFuzzError::BrokenInvariant(case_data)
848                | InvariantFuzzError::Revert(case_data) => {
849                    // Replay error to create counterexample and to collect logs, traces and
850                    // coverage.
851                    match replay_error(
852                        &case_data,
853                        &invariant_contract,
854                        self.clone_executor(),
855                        &self.cr.mcr.known_contracts,
856                        identified_contracts.clone(),
857                        &mut self.result.logs,
858                        &mut self.result.traces,
859                        &mut self.result.line_coverage,
860                        &mut self.result.deprecated_cheatcodes,
861                        progress.as_ref(),
862                        show_solidity,
863                    ) {
864                        Ok(call_sequence) => {
865                            if !call_sequence.is_empty() {
866                                // Persist error in invariant failure dir.
867                                if let Err(err) = foundry_common::fs::create_dir_all(failure_dir) {
868                                    error!(%err, "Failed to create invariant failure dir");
869                                } else if let Err(err) = foundry_common::fs::write_json_file(
870                                    failure_file.as_path(),
871                                    &InvariantPersistedFailure {
872                                        call_sequence: call_sequence.clone(),
873                                        driver_bytecode: Some(test_bytecode.clone()),
874                                    },
875                                ) {
876                                    error!(%err, "Failed to record call sequence");
877                                }
878
879                                let original_seq_len =
880                                    if let TestError::Fail(_, calls) = &case_data.test_error {
881                                        calls.len()
882                                    } else {
883                                        call_sequence.len()
884                                    };
885
886                                counterexample =
887                                    Some(CounterExample::Sequence(original_seq_len, call_sequence))
888                            }
889                        }
890                        Err(err) => {
891                            error!(%err, "Failed to replay invariant error");
892                        }
893                    };
894                }
895                InvariantFuzzError::MaxAssumeRejects(_) => {}
896            },
897
898            // If invariants ran successfully, replay the last run to collect logs and
899            // traces.
900            _ => {
901                if let Err(err) = replay_run(
902                    &invariant_contract,
903                    self.clone_executor(),
904                    &self.cr.mcr.known_contracts,
905                    identified_contracts.clone(),
906                    &mut self.result.logs,
907                    &mut self.result.traces,
908                    &mut self.result.line_coverage,
909                    &mut self.result.deprecated_cheatcodes,
910                    &invariant_result.last_run_inputs,
911                    show_solidity,
912                ) {
913                    error!(%err, "Failed to replay last invariant run");
914                }
915            }
916        }
917
918        self.result.invariant_result(
919            invariant_result.gas_report_traces,
920            success,
921            reason,
922            counterexample,
923            invariant_result.cases,
924            invariant_result.reverts,
925            invariant_result.metrics,
926            invariant_result.failed_corpus_replays,
927        );
928        self.result
929    }

931    /// Runs a fuzzed test.
932    ///
933    /// Applies the before test txes (if any), fuzzes the current function and returns the
934    /// `TestResult`.
935    ///
936    /// Before test txes are applied in order and state modifications committed to the EVM database
937    /// (therefore the fuzz test will use the modified state).
938    /// State modifications of before test txes and fuzz test are discarded after test ends,
939    /// similar to `eth_call`.
940    fn run_fuzz_test(mut self, func: &Function) -> TestResult {
941        // Prepare fuzz test execution.
942        if self.prepare_test(func).is_err() {
943            return self.result;
944        }
945
946        let runner = self.fuzz_runner();
947        let fuzz_config = self.config.fuzz.clone();
948
949        let progress = start_fuzz_progress(
950            self.cr.progress,
951            self.cr.name,
952            &func.name,
953            fuzz_config.timeout,
954            fuzz_config.runs,
955        );
956
957        // Run fuzz test.
958        let fuzzed_executor =
959            FuzzedExecutor::new(self.executor.into_owned(), runner, self.tcfg.sender, fuzz_config);
960        let result = fuzzed_executor.fuzz(
961            func,
962            &self.setup.fuzz_fixtures,
963            &self.setup.deployed_libs,
964            self.address,
965            &self.cr.mcr.revert_decoder,
966            progress.as_ref(),
967        );
968        self.result.fuzz_result(result);
969        self.result
970    }

972    /// Prepares single unit test and fuzz test execution:
973    /// - set up the test result and executor
974    /// - check if before test txes are configured and apply them in order
975    ///
976    /// Before test txes are arrays of arbitrary calldata obtained by calling the `beforeTest`
977    /// function with test selector as a parameter.
978    ///
979    /// Unit tests within same contract (or even current test) are valid options for before test tx
980    /// configuration. Test execution stops if any of before test txes fails.
981    fn prepare_test(&mut self, func: &Function) -> Result<(), ()> {
982        let address = self.setup.address;
983
984        // Apply before test configured functions (if any).
985        if self.cr.contract.abi.functions().filter(|func| func.name.is_before_test_setup()).count()
986            == 1
987        {
988            for calldata in self.executor.call_sol_default(
989                address,
990                &ITest::beforeTestSetupCall { testSelector: func.selector() },
991            ) {
992                // Apply before test configured calldata.
993                match self.executor.to_mut().transact_raw(
994                    self.tcfg.sender,
995                    address,
996                    calldata,
997                    U256::ZERO,
998                ) {
999                    Ok(call_result) => {
1000                        let reverted = call_result.reverted;
1001
1002                        // Merge tx result traces in unit test result.
1003                        self.result.extend(call_result);
1004
1005                        // To continue unit test execution the call should not revert.
1006                        if reverted {
1007                            self.result.single_fail(None);
1008                            return Err(());
1009                        }
1010                    }
1011                    Err(_) => {
1012                        self.result.single_fail(None);
1013                        return Err(());
1014                    }
1015                }
1016            }
1017        }
1018        Ok(())
1019    }

1021    fn fuzz_runner(&self) -> TestRunner {
1022        let config = &self.config.fuzz;
1023        let failure_persist_path = config
1024            .failure_persist_dir
1025            .as_ref()
1026            .unwrap()
1027            .join(config.failure_persist_file.as_ref().unwrap())
1028            .into_os_string()
1029            .into_string()
1030            .unwrap();
1031        fuzzer_with_cases(
1032            config.seed,
1033            config.runs,
1034            config.max_test_rejects,
1035            Some(Box::new(FileFailurePersistence::Direct(failure_persist_path.leak()))),
1036        )
1037    }

1049fn fuzzer_with_cases(
1050    seed: Option<U256>,
1051    cases: u32,
1052    max_global_rejects: u32,
1053    file_failure_persistence: Option<Box<dyn FailurePersistence>>,
1054) -> TestRunner {
1055    let config = proptest::test_runner::Config {
1056        failure_persistence: file_failure_persistence,
1057        cases,
1058        max_global_rejects,
1059        // Disable proptest shrink: for fuzz tests we provide single counterexample,
1060        // for invariant tests we shrink outside proptest.
1061        max_shrink_iters: 0,
1062        ..Default::default()
1063    };
1064
1065    if let Some(seed) = seed {
1066        trace!(target: "forge::test", %seed, "building deterministic fuzzer");
1067        let rng = TestRng::from_seed(RngAlgorithm::ChaCha, &seed.to_be_bytes::<32>());
1068        TestRunner::new_with_rng(config, rng)
1069    } else {
1070        trace!(target: "forge::test", "building stochastic fuzzer");
1071        TestRunner::new(config)
1072    }
1073}

1075/// Holds data about a persisted invariant failure.
1076#[derive(Serialize, Deserialize)]
1077struct InvariantPersistedFailure {
1078    /// Recorded counterexample.
1079    call_sequence: Vec<BaseCounterExample>,
1080    /// Bytecode of the test contract that generated the counterexample.
1081    #[serde(skip_serializing_if = "Option::is_none")]
1082    driver_bytecode: Option<Bytes>,
1083}
1084
1085/// Helper function to load failed call sequence from file.
1086/// Ignores failure if generated with different test contract than the current one.
1087fn persisted_call_sequence(path: &Path, bytecode: &Bytes) -> Option<Vec<BaseCounterExample>> {
1088    foundry_common::fs::read_json_file::<InvariantPersistedFailure>(path).ok().and_then(
1089        |persisted_failure| {
1090            if let Some(persisted_bytecode) = &persisted_failure.driver_bytecode {
1091                // Ignore persisted sequence if test bytecode doesn't match.
1092                if !bytecode.eq(persisted_bytecode) {
1093                    let _= sh_warn!("\
1094                            Failure from {:?} file was ignored because test contract bytecode has changed.",
1095                        path
1096                    );
1097                    return None;
1098                }
1099            };
1100            Some(persisted_failure.call_sequence)
1101        },
1102    )
1103}

1105/// Helper functions to return canonicalized invariant failure paths.
1106fn invariant_failure_paths(
1107    config: &InvariantConfig,
1108    contract_name: &str,
1109    invariant_name: &str,
1110) -> (PathBuf, PathBuf) {
1111    let dir = config
1112        .failure_persist_dir
1113        .clone()
1114        .unwrap()
1115        .join("failures")
1116        .join(contract_name.split(':').next_back().unwrap());
1117    let dir = canonicalized(dir);
1118    let file = canonicalized(dir.join(invariant_name));
1119    (dir, file)
